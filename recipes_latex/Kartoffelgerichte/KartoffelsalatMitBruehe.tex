\subsection{Kartoffelsalat mit Brühe (4 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 1kg Kartoffeln (festkochend)
	\item 1 Zwiebeln (klein)
	\item 4 EL Sonnenblumenöl
	\item 1 EL Senf (mittelscharf)
	\item 350ml Gemüsebrühe
	\item 5-6 EL Weißwein Essig
	\item 80g gewürfelter, roher Schinken
	\item Salz
	\item Pfeffer
	\item 1 Bund Schnittlauch
	\item Zucker
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Kartoffeln waschen und in Salzwasser ca. 20 Minuten garen
	\item Währenddessen: Zwiebeln in kleine Würfel schneiden
	\item Öl in einem Topf erhitzen und die Zwiebeln darin ca. 15 Minuten zugedeckt darin dünsten
	\item Senf und die Brühe dazugeben und aufkochen
	\item Essig und Schinken dazugeben
	\item die Brühe dann kräftig mit Salz und Pfeffer würzen
	\item Kartoffeln abgießen, pellen und in Scheiben schneiden
	\item Kartoffeln für mind. 20 Minuten in der Brühe ziehen lassen
	\item Schnittlauch waschen und schneiden
	\begin{description}[noitemsep,nolistsep]
		\item $\rightarrow$ schräg in feine Ringe schneiden 
	\end{description}
	\item Schnittlauch zum Salat geben
	\item Kartoffelsalat mit Salz, Pfeffer und Zucker abschmecken
\end{enumerate}
