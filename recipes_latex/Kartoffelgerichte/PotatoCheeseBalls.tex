\subsection{Potato Cheese Balls (need testing)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 200g Kartoffeln
	\item Wasser (zum Kochen der Kartoffeln)
	\item 1 Prise Salz
	\item 1 TL Mehl
	\item 1 Pck. Mozzarellakäse (Kugeln)
	\item 1 Ei
	\item 20ml Milch
	\item 50g Mehl
	\item Brotkrümmel
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Kartoffeln schälen
	\item Kartoffeln 20 Minuten lang dampfen
	\item Kartoffeln in eine Schüssel geben und zerdrücken, solange die noch heiß sind
	\item Salz und Mehl dazugeben
	\item Mozzarella vom Teig umschlingen lassen und kleine Kugeln formen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Vergleich: Stein in Schneeball verstecken
	\end{description}
	\item Kugeln 30 Minuten im Kühlschrank lagern
\end{enumerate}
\begin{enumerate}[noitemsep]
\setcounter{enumi}{6}
	\item Ei und Milch in einer Schüssel verrühren
	\item Mehl dazugeben und mischen
	\item Kartoffeln in der Flüssigkeit wälzen und dann in den Krümmeln wälzen (1-2x)
	\item 3-5 Minuten in einer heißen Pfanne frittieren bis sie goldbraun sind
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ währenddessen: mal hier und dort drehen, um Verbrennungen zu meiden
	\end{description}
	\item Öl beim Herausnehmen abtropfen lassen
\end{enumerate}
