\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item Zitronen
	\item Honig/Zucker (vorzugsweise Honig)
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Glas desinfizieren mit kochendem Wasser
	\item Zitronen waschen und in Scheiben schneiden
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow ca. 5mm dick
		\item \rarrow Kerne entfernen
		\item \rarrow Die Enden der Zitronen aufheben
	\end{description}
	\item 3 Scheiben Zitrone in das Glas geben
	\item Gebe ca. 1 EL Honig auf die Zitronenscheiben
	\item Wiederhole die letzten beiden Schritte bis das Glas voll ist
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Sollte Zucker verwendet werden: Fülle bei der letzten Charge den Zucker hoch bis zum Rand
	\end{description}
	\item Eine Schicht Frischhaltefolie über das Glas legen, sodass es direkten Kontakt mit der obersten Zitrone hat
	\item Das Ende der Zitrone als Deckel nun auf die Folie legen und alles etwas runter drücken 
	\item Mit der überstehenden Folie das Ende der Zitrone einwickeln und und das Glas schließen
	\item Glas für 1 Woche in den Kühlschrank stellen
\end{enumerate}
