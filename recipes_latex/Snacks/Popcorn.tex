\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 25g Butter
	\item 150g Popcorn-Körner
	\item 100-150g Zucker
	\item 1 Prise Salz
	\item 25g Butter
	\item 4 EL Öl
	\item 1 Schuss Wasser
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Öl in einen Topf geben und Popcorn-Körner gleichmäßig verteilen 
	\item Butter dazugeben und Öl heiß werden lassen
	\item Sobald das Öl heiß geworden ist, Herd auf mittlere Stufe stellen
	\begin{description}[noitemsep, nolistsep]
		\item Kontinuierlich mit Stäbchen rühren
	\end{description}
	\item Sobald die ersten Körner poppen: Deckel drauf
	\item Sobald keine Körner mehr poppen: den ganzen Topf schütteln
	\item In einer heißen Pfanne die Butter zerschmelzen 
	\item Zucker und Wasser dazugeben und alles gut umrühren
	\item Sobald der Zucker Farbe annimmt: Herd ausschalten und rühren bis der Zucker braun wird
	\item Karamell zum Popcorn hinzufügen
	\item Deckel drauf legen und alles gründlich schütteln
\end{enumerate}
