\label{Fruchtgummis}
\paragraph{Zubereitung}
\begin{itemize}[noitemsep]
	\item 250g Früchte
	\item 7g Agar Agar
	\item 1 EL Zucker
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Zutaten mit einem Mixer pürieren
	\item Masse aufkochen
	\item Masse in Eiswürfelformen geben
	\begin{description}[noitemsep,nolistsep]
		\item \rarrow Formen können mit Öl eingeschmiert werden, damit sich diese leichter aus der Form nehmen lassen
	\end{description}
	\item Eiswürfelformen in das Tiefkühlfach geben
\end{enumerate}
