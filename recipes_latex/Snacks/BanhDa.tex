\subsection{Bánh Đa (need testing)}
\selectlanguage{ngerman}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 200g Reismehl
	\item 100g Tapiokastärke
	\item 1 TL Salz
	\item 1 TL Zucker 
	\item 50g Sesam (schwarz)
	\item 400ml Kokosmilch
	\item 150ml Wasser 
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Kokosmilch-Dose 10 Minuten in kochendem Wasser erhitzen
	\item alle Zutaten (mit Ausnahme des Sesams) in einer Schüssel miteinander vermischen
	\item Behälter abdecken und Mischung 2h lang ziehen lassen
	\item Danach: Mischung gut umrühren und den Sesam hinzufügen
	\item Sesam gründlich unterrühren
	\item eine Suppenkelle von der Mischung in einer großen Pfanne verteilen (Herdstufe 2)
	\begin{description}[nolistsep, noitemsep]
		\item $\rightarrow$ damit der Teig einfacher verteilt wird, kann die gesamte Pfanne rotiert werden
	\end{description}
	\item 1-2 Minuten warten 
	\item sobald die Konsistenz \glqq Wrapartig\grqq{} ist: fertig
	\item 1-2 Tage in der Sonne trocknen lassen
	\item Ofen auf 55°C vorheizen
	\item die "Wraps" für 4h in den Ofen legen
	\item Ofen auf 65°C steigern
	\item \glqq Wraps\grqq{} weitere 2h warten lassen
	\vspace{0.5cm}
	\item 7 Minuten in einem 180° Ofen oder einer Heißluftfritteuse aufwärmen
\end{enumerate}
