\subsection{Chips (20 Minuten)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 2 große Kartoffeln
	\item 1 TL Salz
	\item Gewürze (z.B. Paprikapulver, Currypulver, Chili, Oregano (optional)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Kartoffeln in dünne Scheiben schneiden
	\item Kartoffeln waschen
	\item in einer Siebschüssel abtropfen lassen
	\item Chips auf einem Tuch verteilen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ keine Überlappungen
	\end{description}
	\item Auf dieses Tuch die die Chips ein weiteres Tuch drauflegen 
	\item dieses Tuch anschließend leicht drücken
	\item Pfanne mit Öl bereit machen
	\item Sobald Öl heiß: Kartoffelscheiben einfügen (nicht voll)
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ 2 Minuten (bis die gold werden)
		\item $\rightarrow$ "almost a little bit brown"
	\end{description}
	\item Kartoffeln würzen (optional)
	\item Ofen auf 120\textdegree C und Chips rein
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Länge kommt auf die Dicke der Scheiben an
		\item $\rightarrow$ ca. 10-15 Minuten
	\end{description}
	\item zusätzlich würzen (optional)
\end{enumerate}
