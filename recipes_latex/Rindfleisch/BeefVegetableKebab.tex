\subsection{Grilled Beef Vegetable Kebab (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 1 Zwiebel
	\item 2 EL Rosmarin 
	\item 6 Zehen Knoblauch
	\item 2 TL Zitronenabrieb einer Zitrone
	\item 1 Schuss Zitronensaft
	\item 80ml Rinderbrühe
	\item 5 EL Pflanzenöl
	\item 3 Prisen Salz 
	\item 1 Prise Zucker (weiß)
	\item Pfeffer (schwarz)
	\item 3 EL Tomatenmark
	\vspace{0.5cm}
	\item 900g Rinderlende
	\item 1 Zucchini (groß)
	\item 2 Paprika (je 1 rot und 1 grün)
	\item 1 Zwiebel (rot, groß)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Knoblauch und Zwiebel grob schneiden und mit den restlichen Zutaten der Marinade in einem Mixer zu einer Marinade verarbeiten
	\item Fleisch in ca. 5cm-Würfel Stücke schneiden
	\item Zucchini längst schneiden und in 2.5cm-Segmente schneiden
	\item Paprika entkernen und in in 1.5-3.5 Zentimeter lange Stücke schneiden
	\item rote Zwiebel halbieren, den Kern entfernen, zu Wedges schneiden und diese dann dritteln/ vierteln
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ es sollen 1.5cm Stücke entstehen 
	\end{description}
	\item Gemüse in eine separate Schüssel geben
	\item 3/4 der Marinade zum Fleisch geben
	\item 1/4 zu dem Gemüse geben 
	\item Fleisch für 1h im Kühlschrank und Gemüse für 1h bei Zimmertemperatur ziehen lassen
	\item Danach: Fleisch und Gemüse auf verschiedene Spieße stecken
	\item Gemüse und Fleisch auf einem vorgeheizten Grill für 6 Minuten grillen, bevor man sie wendet 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Gemüse bei mittlerer Hitze
		\item $\rightarrow$ Fleisch bei hoher Hitze
	\end{description}
	\item für weitere 6 Minuten grillen 
	\item Fleisch rausnehmen und das Gemüse für weitere 5-6 Minuten, bis sie leicht schwarz und weich werden 
\end{enumerate}
