\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 100ml Wasser
	\item 1kg Hackfleisch (gemischt)
	\item 20g Salz
	\item 2 TL Pfeffer (weiß)
	\item 3 Zehen Knoblauch
	\item 1 Zwiebel
	\item 1g Natron
	\item Rinderbrühe
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Wasser in einen Topf geben
	\item Knoblauch grob schneiden und inkl. Schale im Topf kurz aufkochen lassen
	\item Nachdem das Knoblauchwasser abgekühlt ist durch einen Sieb inkl. des Pfeffers und des Salzes zum Fleisch geben
	\item Hackfleisch gut durchkneten (Knetmaschine für 30 Minuten)
	\item Fleisch mit Frischhaltefolie abdecken und für 24h in den Kühlschrank stellen
	\item Hackfleisch in Würste formen (im Video wurde ein Wurstfüller benutzt)
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow ungefähr so groß wie ein Daumen
		\item \rarrow 20-25g pro Stück
	\end{description}
	\item Cevapcici auf Holzkohle grillen und in regelmäßigen Abständen mit Rinderbrühe besprühen während sie gerade noch grillen
\end{enumerate}
