\subsection{Steak}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 300g Steak
	\item Salz und Pfeffer
	\item Pflanzenöl zum Braten
	\item 3 Zehen Knoblauch
	\item 1/2 Handvoll Thymian
	\item 1 EL Butter (gehäuft)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Knoblauch \glqq{}aufknacken\grqq{}
	\item Steak beidseitig kräftig mit Salz und Pfeffer bestreuen
	\item Öl auf hohe Hitze vorwärmen
	\item Steak reingeben
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Referenz für die Hitze des Öls: im Normalfall würde der Feuermelder angehen
	\end{description}
	\item Steak für eine Minute braten
	\item Steak wenden und wieder zu Schritt 4 springen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ insgesamt 5 mal
	\end{description}
	\item direkt nach dem 5. Wenden: Butter, Knoblauch und Thymian rein und eine Minute braten
	\item wiederholt die Soße auf das Steak löffeln
	\item Sobald die Minute rum ist: fertig
\end{enumerate}
