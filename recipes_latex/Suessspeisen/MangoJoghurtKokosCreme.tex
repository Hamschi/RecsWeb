\subsection{Kokos-Mango-Joghurtcreme (2 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep, nolistsep]
	\item 200g griechischer Joghurt
	\item 1 reife Mango
	\item 2 TL Zitronensaft (frisch gepresst)
	\item 2 TL Kokosblütenzucker (habe erst mit gewöhnlichem Zucker getestet)
	\item 2 Tropfen Vanillepaste (habe erst mit Vanillezucker getestet)
	\item 2 TL Kokosraspeln
	\item 4 Kekse Spritzgebäck
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Mango halbieren und aus denen Mango-Igel formen
	\item Aus den Mango-Igeln Mangowürfel ausschneiden und die schönsten Mangowürfel als Deko aufheben
	\item den Rest mit Zitronensaft in einen Stabmixer geben und fein pürieren
	\item 1/3 des Mangopürees, Zucker, Vanillepaste und die Kokosraspeln mit dem Joghurt verrühren
	\item Spritzgebäck zerkleinern
	\item (fast) die Hälfte aus der Kekskrümel in ein (größeres) Glas schütten
	\item die Keksschicht mit einer Joghurtschicht bedecken
	\item mit der Hälfte des Mangopürees eine weitere Schicht bilden
	\item eine weitere Joghurtschicht bilden
	\item die hübschen Mangowürfe obenauf drapierne und diese mit den Kekskrümeln toppen
	\begin{description}[noitemsep, nolistsep]
		\vspace{0.3cm}
		\item \textbf{Achtung:} die Zutaten sollten gleichmäßig auf zwei Gläser verteilt werden
	\end{description}	 
\end{enumerate}
