\subsection{Schokocrossies (need pic)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 1 EL Palmin
	\item 5 Tafeln Schokolade (zartbitter)
	\item 5 Tafeln Schokolade (Vollmilch)
	\item Cornflakes
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item die Schokolade in kleine Stücke brechen
	\item das Palmin in einem großen Topf zum Schmelzen bringen
	\item Danach: die Schokolade in den Topf geben und zum Schmelzen bringen
	\item Danach: Cornflakes in den Topf schütten, bis sich (fast) alles Schokolade an Cornflakes angeheftet hat
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ dabei kontinuierlich alles verrühren
	\end{description}
	\item zwei Teelöffel nehmen und mit denen Schokocrossie-Haufen bilden und auf ein Backblech legen
	\item die Schokocrossies kühlen lassen bis die Schokolade wieder fest wird
\end{enumerate}
