\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500g Erdbeeren
	\item 4 EL Puderzucker (gestrichen)
	\item 4 Eigelbe (Größe M)
	\item 4 EL Mascarpone
	\item Minze oder Basilikum zum Verzieren
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Erdbeeren waschen, putzen und vierteln
	\item Puderzucker durch einen Sieb geben
	\item Eigelbe und Puderzucker mit einem elektrischen Rührgerät einige Minuten schaumig schlagen
	\item Mascarpone glattrühren und anschließend unter die Eigelb-Zuckermischung heben
	\item  Ofen auf der Grillstufe vorheizen
	\item Erdbeeren in einen Teller geben und mit der Creme übergießen
	\begin{description}[noitemsep,nolistsep]
		\item \rarrow Teller sollte etwas Hitze vertragen!
	\end{description}
	\item Teller im Ofen möglichst dicht unter die Heizstäbe stellen und für etwa 2-3 Minuten gratinieren
	\begin{description}[noitemsep,nolistsep]
		\item \rarrow Alternativ: Bunsenbrenner
	\end{description}
\end{enumerate}
