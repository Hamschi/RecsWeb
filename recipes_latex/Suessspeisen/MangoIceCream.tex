\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 300g Mango (gewürfelt, gefroren)
	\item 4 EL Honig
	\item 120ml gr. Joghurt
	\item 1-3 TL Limettensaft
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Backform aus Metall ins Tiefkühlfach legen
	\item Joghurt, Limettensaft und Honig in einen Püriermixer tun und pürieren
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Honig mittig vom Pürierer tun, damit der klebende Honig nicht an die Wände gerät
	\end{description}
	\item Mangowürfel darin pürieren bis eine glatte Masse entsteht
	\item Entweder frisch aus dem Pürierer servieren oder für eine Stunde in der Backform ins Gefrierfach legen
	\item 
\end{enumerate}
