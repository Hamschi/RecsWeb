\subsection{Creme Charamel}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item Nước Đường
	\item 350ml Vollmilch
	\item 120ml Kondensmilch (süß) 
	\item 3 Eier
	\item 1/2(?)	 TL Vanilleextrakt 
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Nước Đường \selectlanguage{ngerman} zubereiten und nach dem Fertigstellen in Souffle-Förmchen kippen und darin gleichmäßig verteilen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ \glqq{}Wasserspiegel\grqq{} bei 1mm
	\end{description}
	\item die drei Eier in einer Schüssel schlagen 
	\item Die Kondensmilch gefolgt von der Vollmilch in einen Topf schütten und verrühren
	\item erhitze die Milch
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ sie soll aber nicht kochen
		\item $\rightarrow$ dabei kontinuierlich rühren
	\end{description}
	\item Die Milch langsam zu den Eiern geben und dabei kontinuierlich rühren 
	\item das Vanilleextrakt unterrühren 
	\item die Flüssigkeit durch einen Sieb gehen lassen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ eine Kelle hilft dabei sehr!
	\end{description}
	\item kippe die Flüssigkeit nun in die Förmchen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ die Flüssigkeit ist zu diesem Zeitpunk noch leicht warm  
	\end{description}
	\item die Förmchen mit Alufolie abdecken 
	\item Die Förmchen auf ein kleines Blech legen und das Blech zur Hälfte mit kochendem Wasser befüllen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ sie hat auf ein kleines Blech nur zwei Förmchen gehabt 
	\end{description}
	\item nun das Backblech in den Ofen tun und für 40 Minuten bei 150-160\textdegree C backen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ man kann einen Zahnstocher benutzen, um sich der Konsistenz zu gewissern
		\item $\rightarrow$ wenn man die Form schüttelt und der Flan sich nicht bewegt, ist die gewünschte Konsistenz erreicht 
	\end{description}
	\item den Flan nun abkühlen lassen 
\end{enumerate}
