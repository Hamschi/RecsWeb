\subsection{Anko - Koshian (need pic)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 250g Azukibohnen
	\item 1-1,25l Wasser 
	\item 1-1,25l Wasser
	\item 200g-250g Zucker
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Bohnen 5 Minuten im Wasser kochen lassen
	\item Bohnen in einen Seiher(?)/Sieb schütten und das Wasser entfernen
	\item Bohnen in einen Topf mit Wasser legen
	\item Topfdeckel drauf und 1,5-2 Stunden köcheln lassen
	\begin{description}[noitemsep, nolistsep]
		\item alternativ: 20 Minuten in einem Dampfkochtopf kochen lassen
		\item $\rightarrow$ \glqq until the beans are very soft\grqq{}
	\end{description}
	\item Bohnen in einen Food Processor 2 Minuten lang mischen(?) bis eine glatte Flüssigkeit entsteht
	\item Flüssigkeit durch einen feinen Sieb passieren lassen
	\item Paste in einen Topf schütten und Zucker hinzugeben
	\item bei mittlerer Temperatur kochen lassen
	\item Währenddessen: 15 Minuten lang konstant rühren
	\begin{description}[noitemsep, nolistsep]
		\item Resultat: sehr dickflüssig und glänzend
	\end{description}
	\item auf einen flachen Behälter (wie etwa ein Backblech) verteilen und kühlen lassen
\end{enumerate}
