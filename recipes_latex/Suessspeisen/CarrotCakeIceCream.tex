\subsection{Carrot Cake Ice Cream (8 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 2 EL Butter
	\item 4-5 Karotten
	\item 180g Zucker (braun)
	\item 1/2 TL Salz
	\item 1,25 TL Zimt
	\item 113g Frischkäse
	\item 400g Kondensmilch (süß)
	\item 1 TL Vanilleextrakt
	\item 600ml Sahne
	\item 100g verschiedene Nüsse (gehackt)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Bringe die Butter in einem Topf zum schmelzen und kombiniere darin die geschmolzene Butter mit den Karotten und dem Zucker
	\item Solange kochen bis die Karotten weich werden (ca. 5 Minuten)
	\item Danach: Zimt untermischen und für eine weitere Minute kochen
	\item Herd ausmachen und die Mischung abkühlen lassen
	\item Währenddessen: Vermenge die Kondensmilch, den Frischkäse und das Vanilleextrakt 
	\item die Karottenmischung zu dieser Mischung hinzufügen und dann in den Kühlschrank stellen 
	\item in einer kühlen Metallschüssel die Schlagsahne mit einem Mixer steif schlagen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ man kann die Metallschüssel vorher in den Kühlschrank/ Gefrierfach legen oder man kann die Metallschüssel in einen Topf mit Eis und Salz legen
	\end{description}
	\item Füge die abgekühlte Karotten-Cream-Cheese Mischung hinzu und verrühre dies zu einer gleichmäßigen Masse
	\item unterrühre die verschiedenen, gehackten Nüsse
	\item Schütte die Eismixtur in einen Behälter, lege Frischhaltefolie direkt auf das Eis und lasse es über Nacht kühlen
\end{enumerate}
