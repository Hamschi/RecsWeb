\subsection{Thai Iced Tea (1 Portion)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 3 EL Thai Tea Mix
	\item 1 cup Wasser (gekocht)
	\item 1.5 EL Kondensmilch (gezuckert)
	\item 2 TL Zucker
	\item 1 Prise Salz
	\item Ein Trinkglas Eis
	\item 2 EL Kondensmilch (ungezuckert)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item ein Geschirrtuch (oder ein Zewa) über einen kleinen/mittelgroßen Behälter legen und darin die Teeblätter kochen
	\item gekochtes Wasser durch dieses Tuch schütten und 3-5 Minuten den Tee ziehen lassen
	\item Kondensmilch (gezuckert), Zucker, Salz und den Tee in ein Glas schütten und alles verrühren
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ wichtig: an dieser Stelle abschmecken. Es sollte etwas süßer sein als das Ideal, weil das Eis das Getränk etwas fader machen wird
	\end{description}
	\item Tee zu dem Glas voll mit Eis geben
	\item Kondensmilch (ungezuckert) hinzufügen
\end{enumerate}
