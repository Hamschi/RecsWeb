\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 2 Cups Mango
	\item 1 Cup Joghurt 
	\item 1/2 Cup Milch
	\item 3 EL Zucker
	\item 1/2 TL Kardamompulver
	\item Minze
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Joghurt, Mango, Milch und Zucker in einem Mixer pürieren 
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Zuckermenge hängt von der Süße der Mango ab
	\end{description}
	\item Füge Kardamompulver hinzufügen und püriere es mit
	\item Sofern die Konsistenz zu dickflüssig ist: füge etwas Wasser hinzu 
\end{enumerate}
