\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 20ml Wasser
	\item 10ml Zuckersirup (z.B. Vanillesirup, alternativ 3 TL Zucker)
	\item 5g Matchapulver
	\item 170ml Milch
\end{itemize}


\paragraph{Zubereitung
\begin{enumerate}[noitemsep]
	\item Wasser und Zuckersirup in einen Becher geben
	\item Matchapulver über einen feinen Sieb in den gleichen Becher geben 
	\item Vorsichtig in kreisenden Bewegungen schlagen, bis die Mischung seidig und ohne sichtbare Klumpen und Blasen ist
	\item  Milch in ein Milchkännchen geben und auf 55\textdegree dampfen lassen 
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Falls nicht vorhanden: in der Mikrowelle oder im Topf erhitzen
	\end{description}
	\item Kippe ein wenig von dieser Milch in den Becher und mische sanft 
	\item Den Rest als schöne Latte-Art reinkippen 
\end{enumerate}
