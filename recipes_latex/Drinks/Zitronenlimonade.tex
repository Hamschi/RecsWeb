\subsection{Zitronenlimonade}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item Zitronen 
	\item Wasser
	\item Zucker
	\begin{description}[noitemsep]
		\item $\rightarrow$ alles im 1:1:1 Verhältnis
	\end{description}
	\vspace{0.5cm}
	\item Eiswürfel und Zitronenscheiben (optional)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Zitronen auspressen bis man ca. 500ml Saft gewinnt
	\item das Fruchtfleisch mit Hilfe eines Siebes von dem Saft trennen 
	\item die gleiche Menge an Wasser in einen Topf mit dem Zucker leicht erhitzen und den Zucker darin auflösen lassen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ fertig, sobald nur noch eine klare, durchsichtige Flüssigkeit übrig bleibt 
	\end{description}
	\item Flüssigkeit zu dem Zitronensaft geben
	\item Limonade mit Zucker und Wasser abschmecken
	\item Zitronenscheiben und Eis in die Limonade geben (optional)
	\item alles miteinander vermengen
\end{enumerate}
