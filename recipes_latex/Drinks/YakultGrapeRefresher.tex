\subsection{Yakult Grape Refresher}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item Eiswürfel
	\item Zitronenscheiben
	\item Traubensaft (Mogu-Mogu-Grape)
	\item 1 Flasche Yakult
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Eiswürfel auf den Boden des Glases legen und zwischendrin zwei Scheiben Zitronen und ein Halm Pfefferminz auslegen
	\item Glas mit Mogu voll füllen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Platz für Yakult lassen
	\end{description}
	\item die Flasche Yakult reinschütten
\end{enumerate}
