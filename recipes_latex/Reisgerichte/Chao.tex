\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 1 Cup Reis
	\item 8 Cup Wasser
	\vspace{0.5cm}
	\item 300g Hähnchenfleisch
	\item 2 TL Maisstärke
	\item 1 EL Austernsoße
	\item 2 EL Wasser
	\item 1/2 TL Hühnerbrühepulver
	\item 2 EL Pflanzenöl
	\vspace{0.5cm}
	\item 14g Ingwer (optional)
	\item 1 TL Salz
	\item 1 TL Hühnerbrühepulver
	\item Grüner Teil der Frühlingszwiebeln
	\item Pfeffer
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Ingwer in dünne Streifen schneiden, Frühlingszwiebeln in (SEHR) dünne Rohre
	\item Hähnchen in \textbf{sehr} dünne Streifen/ Fäden schneiden 
	\item Fleisch mit den Gewürzen in der mittleren Liste (s.o) marinieren
	\item Das Wasser in einem Topf zum Kochen bringen
	\item Den Reis waschen, sodass die Stärke weg ist
	\item Sobald das Wasser kocht: Reis gleichmäßig verteilt hinzufügen und mit einem Schneebesen grob umrühren
	\item Topfdeckel drauflegen, sodass noch Luft rein kommen kann und warten, bis das Wasser erneut kocht
	\item Danach: auf mittlerer Hitze für 25 Minuten kochen lassen
	\item Wenn der Reis fluffig aussieht: den Topf 2 bis 3 Minuten lang schnell und konstant mit einem Schneebesen umrühren
	\item Den Herd auf hohe Hitze stellen und das Hähnchenfleisch nach und nach 30 bis 60 Sekunden lang unter ständigem Rühren hinzufügen
	\item Das Hähnchen weitere 1 bis 2 Minuten umrühren
	\item Die Ingwerstreifen sowie Salz und Hühnerbrühe hinzufügen
	\item 20-30s alles umrühren
	\item Mit Frühlingszwiebeln und Pfeffer garnieren und servieren!
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Beim Essen die Frühlingszwiebeln und den Pfeffer in den Reisbrei unterrühren
	\end{description}
\end{enumerate}
