\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item Schwarze Bohnen
	\item 4 Cups Wasser
	\item 8 Cups Wasser
	\item 1.5 Zwiebeln (groß)
	\item 4 Zehen Knoblauch
	\item 2 Avocadoblätter (alternativ: Lorbeerblatt)
	\item 4 Epazoteblätter (alternativ: 3-4 Korianderstängel)
	\item 2 Zwiebeln
	\item Salz
	\vspace{0.5cm}
	\item 1 Dose Dosentomaten
	\item 1 TL Kreuzkümmel
	\item 1/2 Zwiebel (groß)
	\item 2 Zehen Knoblauch
	\item Salz
	\item 400ml Hühnerbrühe
	\vspace{0.5cm}
	\item Reis
	\item Mais, Erbsen, Karotten (gewürfelt), 1 Serrano-Chili, Koriander
	\item Queso Fresco zum Garnieren
\end{itemize}


\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Schwarze Bohnen entfernen, die nicht perfekt sind und entfernen
	\item Bohnen in einem Sieb unter der Spüle waschen
	\item Bohnen in eine Schüssel geben und das Wasser dazugeben, sodass sie komplett abgedeckt sind
	\item Bohnen über Nacht in den Kühlschrank legen
	\item Zwiebeln halbieren, Knoblauch andrücken und inkl. der Bohnen, den Epazoteblättern, den Avocadoblättern und dem Wasser in einen Topf geben und zum Kochen bringen für 60-90 Minuten
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Hin und wieder rühren
		\item \rarrow Bohnen sind fertig, wenn sie sich ohne Widerstand zerdrücken lassen
	\end{description}
	\item Alles mit Ausnahme der Bohnen aus dem Wasser entfernen
	\item Zwiebeln in Würfel schneiden und anschließend in der Pfanne mit einer Prise Salz anbraten
	\item 3 Suppenkellen Bohnen dazugeben (möglichst wenig Wasser) und eine Prise Salz dazugeben
	\item Dosentomaten, Kreuzkümmel, Salz, Zwiebel, Knoblauch und Hühnerbrühe in einen Mixer geben und pürieren 
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Zu dickflüssig: mehr Hühnerbrühe
		\item \rarrow Zu dünnflüssig: mehr Tomaten
	\end{description}
	\item Reis waschen und in einem Topf mit Öl bei mittlerer Hitze anbraten, sodass es leicht braun wird
	\item Danach:  Tomatenflüssigkeit hinzufügen (nicht alles wird benötigt) und umrühren
	\item Sobald es köchelt: Mais, Erbsen, Karotte, Serrano-Chili und Koriander dazugeben und für 12-20 Minuten kochen lassen (mit Deckel)
	\item Danach: Hitze ausschalten
	\item Reis servieren mit Bohnen an der Seite und Käse auf den Bohnen
\end{enumerate}
