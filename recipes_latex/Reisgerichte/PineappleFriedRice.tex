\subsection{Pineapple Fried Rice (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 283g geschälte Garnelen (groß)
	\item 3 TL Sojasoße (hell)
	\item 1 Prise Pfeffer (weiß)
	\item 1 EL Fischsoße
	\item 1 EL Austernsoße
	\item 1 EL Palmzucker (alternativ Honig)
	\item 1/2 TL Currypulver
	\item 1 TL Pfeffer (weiß)
	\item 5 Zehen Knoblauch 
	\item 1/2 EL Ingwer 
	\item 2-3 kleine Chilis (optional)
	\item 1 Ananas
	\item 3 Eier
	\item 1 Prise Salz
	\item 1 Prise Pfeffer (weiß)
	\item 112g Karotten und grüne Bohnen
	\item 33g Cashew Kerne (unbehandelt)
	\item 400g alter Reis
	\item 1-2 Frühlingszwiebeln 
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Garnelen kleinschneiden und in einer kleinen Schüssel mit 1 TL Sojasoße und dem weißen Pfeffer marinieren und zur Seite legen 
	\item in einer anderen Schüssel 2 TL Sojasoße, Fischsoße, Austernsoße, Palmzucker, Currypulver und Pfeffer verrühren 
	\item grüne Bohnen und Karotten kleinschneiden 
	\item Ananas halbieren (symmetrisch) und das Fruchtfleisch entnehmen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ ein 3-spaltiges Raster einschneiden
		\item $\rightarrow$ Ananaswürfel mit einem EL entfernen 
	\end{description}
	\item 150g Ananas kleinschneiden und zur Seite legen 
	\item die Eier mit Salz und Pfeffer schlagen und in einer Pfanne omeletteartig backen und zur Seite legen 
	\item mit neuem Öl die Shrimps 2-3 Minuten braten und zur Seite legen  
	\item mit neuem Öl Knoblauch, Ingwer und Chilli in der Pfanne braten  
	\item Sobald Knoblauch goldbraun: Karotten, die grünen Bohnen und Cashew Kerne hinzugeben 
	\item Nach 30-60s: Kalten Reis dazugeben und den Gewürzmix vom Anfang gleichmäßig über den Reis verteilen und braten, bis der Reis gleichmäßig braun ist 
	 \item Ananas dazugeben 
	 \item nach 30-45s Eier, Garnelen und die Frühlingszwiebeln mitbraten
	 \item Reis in einer Ananashälfte servieren, fertig!
\end{enumerate}
