\subsection{Turkish Chicken Kebabs (4 Portionen) (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 230g griechischer Joghurt (Vollmilch)
	\item 6 Zehen Knoblauch (gerieben)
	\item 2 EL Olivenöl
	\item 2 EL Zitronensaft (frisch)
	\item 2 EL Ketchup (ursprünglich Tomatenmark)
	\item 1 EL Chiliflocken
	\item 1 EL Salz
	\item 1.5 EL Kreuzkümmel
	\item 1 TL Pfeffer (schwarz)
	\item 1 TL Paprikapulver
	\item 1 Prise Zimt
	\item 1.1kg Hähnchenschenkel (ohne Knochen)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Joghurt, Knoblauch, Olivenöl, Zitronensaft, Ketchup, Chiliflocken, Salz, Kreuzkümmel, Pfeffer, Paprikapulver und Zimt in einer großen Schüssel verrühren
	\begin{description}[noitemsep]
		\item $\rightarrow$ es entsteht eine orange-besche Marinade
	\end{description}
	\item Hähnchenschenkel halbieren
	\item Hähnchenschenkel in der Marinade gleichmäßig mit der Marinade bedecken
	\item Das Fleisch direkt mit Frischhaltefolie bedecken und für 2-8 Stunden marinieren
	\item Zwei Metallspieße nehmen und an beide die Fleischstücke stecken
	\item Die Kebabs auf den vorgeheizten Grill legen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Mindestens 3-4 Minuten warten, bevor man das Fleisch wendet, weil Joghurtmarinaden gerne kleben bleiben
	\end{description}
	\item drehe den Spieß und grille für weitere 3-4 Minuten 
	\item grille und wende das Fleisch weiter, bis es von Innen nicht mehr rosa ist
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ ca. weitere 6 Minuten
		\item $\rightarrow$ ein Fleischthermometer ins Innere würde mit ca 74\textdegree C reagieren
	\end{description}
\end{enumerate}
