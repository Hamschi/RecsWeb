\titlelink{Hähnchenspieße auf Curryreis (2 Portionen) }{https://www.youtube.com/watch?v=NOQ_78_9UK8&ab_channel=CALLEkocht-OmasRezepte}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 2 Hähnchenbrüste
	\item 1 Paprika
	\item 1 Gemüsezwiebel
	\item 250g Reis (gekocht)
	\item 1 EL Currypulver
	\item 1/2 TL Chili
	\item Salz und Pfeffer
	\item Öl
	\item 0.1l Sahne
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Paprika Deckel und Boden abschneiden und den Hauptkörper sowie die Zwiebel und das Hähnchen in Schachlik-große Stücke schneiden
	\begin{description}[noitemsep, nolistsep	]
		\item $\rightarrow$ bisschen Zwiebel übrig lassen für den Reis
	\end{description}
	\item Ananas auch passend schneiden (sollen nicht zu schmal werden)
	\begin{description}[noitemsep, nolistsep	]
		\item $\rightarrow$ bisschen übrig lassen für den Reis
	\end{description}
	\item Währenddessen: Paprika und Zwiebeln in Salzwasser blanchieren (das geht schnell)
	\item Die Schneidereste klein würfeln
	\item Alles gleichmäßig aufspießen
	\item Spieße mit Salz und Pfeffer würzen
	\item Spieße anbraten, damit sie bisschen Farbe bekommen
	\item Die Spieße rausnehmen und die Schneidereste darin anbraten
	\item Gemüse mit Salz, Pfeffer, Chili und Currypulver anbraten
	\item Sahne dazugeben, gefolgt von dem Hähnchen
	\item das Hähnchen für 160 \textdegree C in den Backofen
	
\end{enumerate}
