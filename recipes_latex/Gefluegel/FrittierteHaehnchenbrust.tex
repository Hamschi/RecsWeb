\subsection{Frittiere Hähnchenbrust}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 1kg Hähnchenbrust
	\item 1/2 Zwiebel (lila)
	\item 1 TL Salz
	\item 1 EL Zucker
	\item 1 TL Paprikapulver
	\item 1 TL Knoblauchpulver/ 1 Zehe Knoblauchzehe
	\item Pfeffer
	\item 1/3 TL MSG
	\item 1 TL Speisestärke (gehäuft)
	\item 4 EL Mehl
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Schenkel herausreißen
	\item Das, was aussieht wie eine Keule, von dem anderen herausreißen
	\item Dreck entfernen
	\item Knochen entfernen
	\item Zwiebel kleinschneiden
	\item Gewürze in der obengenannten Reihenfolge zum Fleisch geben und kneten
	\item Pfanne mit Öl vollmachen (Herdstufe 9)
	\item Fleisch in die Pfanne geben
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Fleisch soll sich nicht überlappen
	\end{description}
	\item sobald es relativ braun ist: wenden
	\item sobald Fleisch goldbraun ist: fertig
\end{enumerate}
