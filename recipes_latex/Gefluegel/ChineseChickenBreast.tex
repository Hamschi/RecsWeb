\titlelink{Chinese Chicken Breast (4 Personen)}{https://www.madewithlau.com/recipes/pan-fried-chicken-breast}

\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 1 EL Salz
	\item 475ml Wasser
	\item 450g Hähnchenbrust
	\item 1 Zwiebel (weiß)
	\item 1 Zwiebel (rot)
	\item 3 EL Maisstärke
	\item 5.5 EL Öl
	\item 1 TL Kochwein
	\item 1 TL Sesamöl
	\vspace{0.5cm}
	\item 2 EL Austernsoße
	\item 2 TL Sojasoße (hell)
	\item 1 EL Kochwein
	\item 1 TL Zucker
	\item 1/2 TL Salz
	\item 1/4 TL Pfeffer (weiß)
	\item 2-3 EL Maisstärke
	\item 1 EL Wasser
	\item 14g Ingwer (gerieben)
	\vspace{0.5cm}
	\item 1 EL Austernsoße
	\item 2 TL Sojasoße (hell)
	\item 2 TL Zucker
	\item 1 TL Hühnerbrühepulver
	\item 1 TL Maisstärke
	\item 3 EL Wasser
	\item 1 EL Kochwein
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Wasser in einen Behälter geben und Salz darin auflösen lassen 
	\item Fleisch rein geben, sodass kein Fleisch über der Wasseroberfläche ist
	\item Nach 30-60 Minuten: Brust in 3 gleichgroße Stücke schneiden entgegengesetzt der Linie
	\item Mit einem 45\textdegree{}-Winkel ca. 2cm-Dicke Scheiben schneiden
	\item Mit der Rückseite des Messers das Messer platt schlagen
	\item Marinade mischen: Austernsoße, Sojasoße, Kochwein, Zucker, Salz, Pfeffer, Maisstärke, Wasser, Ingwer 
	\item Fleisch in die Marinade geben und gleichmäßig kneten
	\item Zwiebeln in Streifen schneiden
	\item Soße zubereiten: Austernsoße, Sojasoße, Zucker, Hühnerbrühepulver, Maisstärke, Wasser, Kochwein mischen
	\item Maisstärke auf einem Teller verteilen
	\item Hähnchen darin wälzen
	\item Bei mittel-hoher Hitze: Die Hähnchen auf einer Seite für 3 Minuten anbraten
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Unterseite soll goldbraun sein und eine Kruste haben
	\end{description}
	\item Für eine weitere Minute braten 
	\item Weiße Zwiebeln gleichmäßig hinzugeben und Hitze etwas reduzieren
	\item Rote Zwiebeln hinzugeben 
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Ein paar Zwiebeln fürs Verzieren übrig lassen
	\end{description}
	\item Kochwein verteilen und die Pfanne für 30s mit einem Deckel abdecken
	\item Soße auf dem Fleisch verteilen und alles schön durchmischen
	\item Sesamöl und die restlichen roten Zwiebeln hinzufügen
	\item Anrichten: Hähnchen auf den Teller anrichten und anschließend die Zwiebeln darauf
\end{enumerate}
