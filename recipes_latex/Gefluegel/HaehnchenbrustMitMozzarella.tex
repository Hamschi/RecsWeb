\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 2 Hähnchenbrüste
	\item 50 Tomaten (getrocknet)
	\item 125g Mozzarella
	\item 150ml Sahne
	\item Salz
	\item Pfeffer
	\item Bisschen Basilikum (optional)
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Hähnchenbrust auf beiden Seiten auf hoher Hitze anbraten, um ihr Farbe zu verleihen
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Pfanne sollte Hitzebeständig sein (Ofen)
	\end{description}
	\item Anschließend entnehmen und die Röststoffe mit der Sahne ablöschen
	\item Hitze reduzieren und Sahne köcheln lassen
	\item Salz und Pfeffer hinzufügen
	\item 5 Taschen schräg in die Hühnerbrust einschneiden
	\item Hühnerbrust von beiden Seiten mit Salz und Pfeffer würzen
	\item In jeder dieser Taschen eine getrocknete Tomate sowie ein Stück Mozzarella reinlegen
	\item Hähnchenbrüste nun die Sahne hineinlegen
	\item Sobald der Mozzarella geschmolzen ist, die Hühnerbrust durch ist und die Sahne Röstfarben angenommen hat: aus dem Ofen entnehmen
	\item (Nur die) Hähnchenbrust auf einem Teller servieren
	\item Die Soße in der Pfanne glattrühren 
	\item Basilikum in dünne Streifen schneiden und unterrühren
	\item Soße neben dem Hähnchen servieren
\end{enumerate}
