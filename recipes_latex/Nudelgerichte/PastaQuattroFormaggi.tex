\subsection{Pasta Quattro Formaggi (2 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 50g Gorgonzola (alternativ Blauschimmelkäse)
	\item 50g Stracchino (alternativ Ricotta oder Hüttenkäse)
	\item 50g Mascarpone
	\item 3-4 EL Pecorino Romano
	\item 200g Penne 
	\item 80ml Vollmilch
	\item 1 EL Salz
	\item 1 Handvoll Walnüsse (gehackt)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Den Strocchino in kleine Stücke zerkleinern
	\item Bereite die Penne zu
	\item Während die Pasta kocht: Pfanne auf mittlerer Hitze erhitzen
	\item Die Milch sowie den Gorgonzola und den Stracchino hinzufügen
	\item Unter Rühren den Käse und die Milch verschmelzen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ das sollte ein langsamer Prozess sein, damit der Käse nicht verbrennt
	\end{description}
	\item Nachdem die Pasta fertig gekocht ist: fülle eine Tasse mit Pastawasser
	\item Die Nudeln abtropfen lassen und füge diese in die Pfanne
	\item füge die Mascarpone sowie 1/3 des Pastawassers in die Pfanne hinzu
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ dabei ständig rühren
	\end{description}
	\item nehme die Pfanne vom Herd und füge den Pecorino Käse hinzu
	\item Rühre gründlich und streue bei der servierten Portion noch Walnüsse drüber
\end{enumerate}
