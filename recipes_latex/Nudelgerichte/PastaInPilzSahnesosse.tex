\subsection{Spaghetti in Pilz-Sahnesoße (2-3 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500g Champignons
	\item 1 Zwiebel
	\item 2 Zehen Knoblauch
	\item 2 Zweige Rosmarin
	\item 2-3 EL Öl
	\item Salz
	\item Pfeffer
	\item Edelsüßpaprika
	\item 2 EL Tomatenmark
	\item 900ml Wasser
	\item 100g Schlagsahne
	\item 350g Spaghetti
	\item 3-4 TL Gemüsebrühe (instant)
	\item 25g Parmesan
	\item Basilikum (zum Garnieren)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Champignons putzen, säubern und je nach Größe halbieren oder dritteln
	\item Zwiebel schälen und fein würfeln, Knoblauch schälen und fein hacken
	\item Rosmarin waschen, trocken schütteln, Nadeln von den Zweigen zupfen und fein hacken
	\item Öl in einem weiten Topf erhitzen und die Champignons unter Wenden darin kräftig anbraten
	\item Zwiebeln und Knoblauch zufügen, kurz anbraten und mit Salz, Pfeffer und Paprika würzen
	\item Tomatenmark zufügen und unter Rühren anschwitzen
	\item mit Wasser und Sahne ablöschen
	\item Spaghetti zufügen, aufkochen, Brühe einrühren und alles zugedeckt bei mittlerer 25-30 Minuten garen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ dabei häufiger umrühren 
	\end{description}
	\item Spaghetti nochmals mit Salz, Pfeffer und Paprikapulver abschmecken 
	\item Parmesan mit einem Sparschäler abziehen 
	\item Nudeln nach Belieben mit Parmesanspänen und Basilikum garnieren
\end{enumerate}
