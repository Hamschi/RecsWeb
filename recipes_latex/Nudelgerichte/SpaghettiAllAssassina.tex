\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 300g Spaghetti
	\item 1-2 Chili (scharf)
	\item 2 Zehen Knoblauch
	\item 1l Wasser
	\item 150g Tomatenmark
	\item 350ml Tomaten (passiert)
	\item Salz und Pfeffer
	\item Olivenöl
	\item Stracciatella di bufala oder Pecorino (optional)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Wasser in einem Topf erhitzen und das Tomatenmark hinzugeben 
	\item Knoblauch und Chili klein schneiden
	\item Olivenöl (großzügig) in einer großen Pfanne verteilen und stark erhitzen
	\item Chili und Knoblauch darin anbraten 
	\item Tomaten (passiert) hinzugeben und braten bis es dicker wird (ca. 2-3 Minuten)
	\item Spaghetti reinlegen, sodass alle den Boden berühren und von Tomatensoße bedeckt sind
	\item Nach 2-3 Minuten: Pasta umdrehen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ darf gerne etwas dunkler werden
	\end{description}
	\item Nach 2-3 Minuten: Nach und nach 3-4 Kellen vom Tomatenwasser dazugeben
	\item Mit Salz und Pfeffer abschmecken
	\item Sobald das Wasser verdunstet: Pasta anbraten lassen und Neues Wasser hinzufügen 
	\item Den vorherigen Schritt solange wiederholen, bis die Pasta Al Dente ist
	\item Am Ende noch einmal anbraten
	\item Beim Servieren mit Olivenöl und dem Käse verzieren
\end{enumerate}
