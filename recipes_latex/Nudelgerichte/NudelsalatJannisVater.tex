\subsection{Sommerlicher Nudelsalat by Jannis' Vater}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500g Hörnchennudeln (alternativ andere, z.B. Fussili)
	\item 6 Tomaten
	\item 5 Spitzpaprika (alternativ 3 Paprika rot)
	\item 1 Salatgurke
	\item 2 Zwiebeln (rot) (alternativ 4 Schalotten)
	\item 1 Ring Fleischwurst
	\item 1 Pck. Schafskäse
	\item 1 Dose Mais
	\item 1 Glas Miracle Whip 
	\item 1 Becher Schmand
	\item 4 EL Balsamico (weiß)
	\item 1 Prise Zucker
	\item Salz, Pfeffer
	\item ggf. Milch oder Wasser
	\item 1 Bund Petersilie
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Nudeln Al dente kochen und kalt werden lassen
	\item Tomaten, Paprika, Gurke, Zwiebeln, Fleischwurst und Käse würfeln und in eine Schüssel geben
	\item Mais und Nudeln in die Schüssel hinzugeben und vermengen
	\item Miracle Whip, Schmand, Balsamico, Zucker, Salz und Pfeffer in eine neue Schüssel geben und gut verrühren
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow ggf. Wasser oder Milch hinzufügen
	\end{description}
	\item Mischung dann über den Salat geben und gut vermengen
	\item Petersilie hacken, über den Salat streuen und gut durchrühren
	\item Nudelsalat kalt stellen
\end{enumerate}
