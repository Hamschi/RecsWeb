\subsection{Spaghetti Bolognese (4 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500g Spaghetti
	\item 500g Hackfleisch (gemischt)
	\item 2 Dosen Tomaten (gehackt)
	\item 2 Karotten
	\item ca. 80g Knollensellerie 
	\item 2 Zwiebeln
	\item 2 Zehen Knoblauch
	\item 1 EL Petersilie (gehackt)
	\item 1 EL Tomatenmark
	\item 4 EL Olivenöl
	\item 250ml Gemüsebrühe
	\item 2 TL Zucker
	\item Salz
	\item Pfeffer
	\item 1 TL Paprikapulver
	\item 1 TL Oregano
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Knollensellerie, Zwiebeln und Karotten in kleine Streifen reiben
	\item Knoblauch und Petersilie fein hacken
	\item Olivenöl im Topf erhitzen und das Hack halbdurch anbraten
	\item Karotten, Sellerie, Petersilie und Knoblauch ca. 5 Minuten lang mitbraten
	\item das Tomatenmark und die Gewürze dazugeben und ca. 1 Minute lang mitbraten
	\item das Fleisch mit den gehackten Tomaten und der Gemüsebrühe ablöschen und bei mittlerer/niedriger Flamme ca. 1h köcheln lassen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ dabei gelegentlich umrühren
	\end{description}
	\item Bolognese zu den fertigen Spaghetti geben und Parmesan über das Ganze reiben
\end{enumerate}
