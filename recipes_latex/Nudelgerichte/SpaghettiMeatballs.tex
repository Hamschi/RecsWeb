\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500g Hackfleisch
	\item 1 Ei
	\item 2 EL Semmelbrösel
	\item 50ml Milch
	\item 10g Parmesan
	\item 1/2 Zwiebel
	\item 1 Handvoll Petersilie
	\item Salz
	\item Pfeffer
	\item Olivenöl zum Braten
	\vspace{0.5cm}
	\item 1/2 Zwiebel
	\item 1 Zehe Knoblauch
	\item Chilliflocken (optional)
	\item 1 Dose Tomaten
	\item Olivenöl zum Braten
	\item Salz
	\item Pfeffer
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Zwiebel in möglichst kleine Würfel schneiden
	\item Zwiebeln im Olivenöl in einer Pfanne glasig dünsten und anschließend zur Seite legen
	\item Semmelbrösel mit der Milch aufweichen
	\item Petersilie grob klein schneiden
	\item Hackfleisch mit Salz und Pfeffer würzen
	\item Ei, Parmesan, Milch und die Zwiebeln zum Fleisch geben und alles gut durchkneten
	\item Fleisch für 30 Minuten in den Kühlschrank stellen
	\item Hände mit Öl einreiben und aus dem Hackfleisch gleichmäßige Kugeln formen
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Eisportionierer funktionieren hierfür super
		\item \rarrow lieber zu groß als zu klein
	\end{description}
	\item Olivenöl in die Pfanne geben und die Meatballs darin anbraten
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Müssen nur von Außen gut aussehen, müssen aber noch nicht durch sein, weil sie später in der Soße fertig garen
	\end{description}
	\item Währenddessen: Zwiebel und Knoblauch klein schneiden
	\item Ggf. etwas Öl abgießen und anschließend die Zwiebel und den Knoblauch anbraten
	\item Sobald die Zwiebeln und der Knoblauch fertig sind: Chiliflocken (optional) und Tomaten hinzugeben
	\item Mit Salz und Pfeffer würzen
	\item Meatballs hinzugeben, Deckel drauf tun und solange simmern lassen, bis die Meatballs gar sind
	\item Nudeln kochen und anschließend direkt in die Pfanne mitgeben und vermengen
	\item Mit Parmesan und Basilikum servieren
\end{enumerate}
