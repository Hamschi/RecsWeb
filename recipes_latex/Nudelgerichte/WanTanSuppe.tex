\subsection{Wan Tan Nudelsuppe (3 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 1/2 Brokkoli
	\item 4 Portionen Instantnudeln
	\item 1l Hähnchenknochenirgendwas
	\item 1 TL Hühnerbrühe
	\item 2 Halme Frühlingszwiebeln
	\item Wan Tans
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Brokkoli in mundgerechte Stücke schneiden
	\item Nudeln kochen
	\item fertige Nudeln auf 3 Schalen aufteilen
	\item Hühnerbrühe in den Topf geben und erhitzen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ ggf. Wasser dazugeben
	\end{description}
	\item Nach 3 minuten dann die Wantans dazugeben und darin kochen
\end{enumerate}
