\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 350-450g Parmesan (Angelo Coassin: 150g)
	\item 350-450g Fettuccine (Angelo Coassin: 360g)
	\item 350-450g Butter (Angelo Coassin: 150g)
\end{itemize}


\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Parmesan fein reiben
	\item Nudeln al dente kochen
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Wasser nur leicht salzen, da der Parmesan das Gericht bereits salzig macht
		\item \rarrow Nicht zu viel Wasser nutzen, damit das Wasser am Ende stärkehaltiger wird
	\end{description}
	\item Sobald die Nudeln fertig sind: direkt aus dem Wasser in eine Glasschüssel umfüllen
	\item Butter in den Nudeln schmelzen lassen
	\item Eine Handvoll Käse nehmen und unterrühren und diesen Schritt so oft wiederholen bis der Käse aufgebraucht ist
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Sollte die Soße zu trocken sein: Nudelwasser hinzufügen
		\item \rarrow Wenn nicht: weiter den Käse hinzufügen
	\end{description}
\end{enumerate}
