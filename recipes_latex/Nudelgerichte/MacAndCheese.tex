\subsection{Mac and Chesse (2 Portionen) (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 200g Hörnchennudeln
	\item 150g Emmentaler  
	\item 150g Bergkäse 
	\item 1/2 Zwiebel (weiß)
	\item 2 EL Butter
	\item 1/2 Becher Sahne
	\item 1 Schuss Worcestershiresauce
	\item Muskatnuss
	\item Salz
	\item Pfeffer
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Zwiebeln in Streifen schneiden 
	\item Butter in der Pfanne zum Schmelzen bringen und die Zwiebeln darin braten bis sie eine goldbraune Farbe annehmen (ca. 8 Minuten)
	\item Währenddessen: Nudeln kochen und Käse reiben
	\item 1 Schuss Sahne und den Käse in die Pfanne mit den Zwiebeln geben 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ eine Handvoll Käse für später aufbewahren 
	\end{description}
	\item Sobald der Käse und die Sahne eine Flüssigkeit geworden sind: Nudeln hinzugeben 
	\item Nudeln mit Muskatnuss, Worcestershiresauce, Salz und Pfeffer abschmecken
	\item Soße gleichmäßig unter den Nudeln verteilen
	\item die Handvoll Käse auf der Pfanne verstreuen
	\item "Im Ofen bei purer Oberhitze gratenieren lassen"
\end{enumerate}
