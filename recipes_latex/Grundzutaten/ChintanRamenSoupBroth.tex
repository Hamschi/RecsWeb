\subsection{Chintan Ramen Soup Broth}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 5g Kombu
	\item 500ml Wasser
	\item 1kg Hähnchenschenkelknochen
	\vspace{0.5cm}
	\item 2l Wasser
	\item 1/2 Zwiebel 
	\item 3 Zehen Knoblauch
	\item 4 Frühlingszwiebeln 
	\item 5g Ingwer 
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Kombu über Nacht im Wasser lassen
	\item Wasser in einem Topf zu den Knochen geben und erhitzen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ entstehenden Dreck dabei entfernen
	\end{description}
	\item Anschließend die Zwiebel, Knoblauch, Frühlingszwiebel, Ingwer hinzugeben und auf niedriger Hitze köcheln lassen 
	\item Nach drei Stunden: Kombu-Wasser hinzugeben (ohne Kombu selbst)
	\item Öl mit einem Löffel oder eine Kelle aus der Suppe entfernen 
	\item Suppe für 6h köcheln lassen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Wasserspiegel soll konstant bleiben durch Hinzufügen von Wasser
	\end{description}
	\item Alles durch einen Sieb geben und auf 20\textdegree Celsius abkühlen lassen
	\item Anschließend in den Kühlschrank legen 
	\item Wenn es serviert werden soll: Suppe wieder aufkochen 
	\item Eignet sich für Shoyu und Shio Ramen
\end{enumerate}
