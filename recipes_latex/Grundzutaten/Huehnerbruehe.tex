\subsection{Hühnerbrühe}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item Knochen von 500g Hähnchen
	\item 1l Wasser
	\item 1 TL Salz 
	\item 1 TL Hühnerbrühe
	\item 1/4 Zwiebel
	\item Ingwer
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item alles in einen Topf geben, Deckel drauf legen und für 30-60 Minuten auf geringer Hitze kochen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ ggf. Dreck vom Fleisch/ von den Knochen ausschöpfen, sobald das Wasser kocht
		\item $\rightarrow$ der Dreck lässt sich leichter ausschöpfen, wenn das Wasser stärker kocht
	\end{description}
\end{enumerate}
