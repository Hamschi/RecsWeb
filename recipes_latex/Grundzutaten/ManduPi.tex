\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 260g Mehl
	\item 1/2 TL Salz
	\item 160ml Wasser 
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Alles in einer Schüssel mit einem Holzlöffel zu einer Masse formen
	\item Mit der Hand den Teig für paar Minuten zu einen festen Klumpen formen
	\item Den Teig für 10-30 Minuten in einer Tüte verschlossen ruhen lassen
	\item Teig für 5-7 Minuten kneten bis es weich und elastisch wird
	\item Teig halbieren und eine Hälfte wieder in der Tüte verschließen
	\item Andere Hälfte in 8 gleichgroße Stücke schneiden. 
	\item Ein Teigstück nehmen und in Scheiben formen (d=14/15cm)
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Kanten sollten dünner sein als die Mitte
		\item \rarrow ungenutzte Teige in der Tüte (getrennt) lassen
	\end{description}
\end{enumerate}
