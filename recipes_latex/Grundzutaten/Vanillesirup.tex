\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 130g Wasser
	\item 100g Rohrzucker
	\item 100g Zucker (weiß)
	\item 1 Schote Vanille
\end{itemize}


\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Das Mark aus der Schote kratzen
	\item Wasser, Zucker, Vanillemark und die Vanilleschote in einen Topf geben und zum Kochen bringen
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Währenddessen regelmäßig umrühren und auflösen lassen
	\end{description}
	\item Sobald die Flüssigkeit kocht: Herd ausmachen
	\item Flüssigkeit in einen Behälter geben und die Schote drin lassen für einen stärkeren Geschmack
	\item Mit Frischhaltefolie die Flüssigkeit direkt abdecken, sodass es auch noch die Flüssigkeit berührt, um Kondensation zu verhindern
	\item Flüssigkeit für drei Tage in den Kühlschrank stellen
	\item Den Sirup sieben und dann in eine Flasche abfüllen	(inkl. Schote)
\end{enumerate}
