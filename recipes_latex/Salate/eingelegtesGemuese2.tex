\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 100g Karotte (geschält)
	\item 100g Kohlrabi (geschält)
	\item 1 TL Salz
	\item 2 EL Zucker
	\item 2 EL Essig
	\item 4 EL Wasser
\end{itemize}


\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Kohlrabi und Karotten in die gewünschte Form schneiden
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Für Bun Cha: ca. 1.5x2cm Scheiben schneiden (ca. 1-2mm dick)
	\end{description}	 
	\item Gemüse in getrennte Schüsseln legen
	\item In jede Schüssel einen halben TL Salz untermischen und alles 30 Minuten ziehen lassen
	\item Danach: Gemüse anquetschen und jegliche Flüssigkeit wegkippen
	\item Alles in einer Schüssel sammeln und den Zucker, den Essig und das Wasser dazugeben und alles gründlich mischen
	\item Alles für 2-3h ruhen lassen  
\end{enumerate}
