\subsection{Corn Ribs (4 Portionen) (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 2 TL Salz
	\item 1 TL Chilipulver
	\item 1 TL Paprikapulver
	\item 1/2 TL Pfeffer
	\item 1/4 TL Kreuzkümmel
	\item 1/4 TL Knoblauchpulver
	\item 1/4 TL Zwiebelpulver
	\item 1/8 TL Cayenne Pfeffer
	\item 4 Mais
	\item 1 EL Pflanzenöl
	\item 4 EL Butter (geschmolzen)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Die oberen 2cm an beiden Enden des Mais abschneiden 
	\item Den Mais längst halbieren
	\item Die Maishälften erneut längst halbieren
	\item Die Gewürze in einer kleinen Schüssel miteinander vermengen
	\item Das Pflanzenöl gleichmäßig auf den Mais verteilen 
	\item Den Mais gleichmäßig mit der Gewürzmischung würzen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ bisschen Gewürz übrig lassen 
	\end{description}
	\item Grillen nach eigenem Geschmack
	\item Mit der Butter beschmieren
\end{enumerate}
