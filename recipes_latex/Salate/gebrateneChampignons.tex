\subsection{Gebratene Champignons (2 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 400g Pilze
	\item 1/4 Zwiebel (lila)
	\item 3 Halme Frühlingszwiebel
	\item 1 Frühlingszwiebelkopf
	\item 1,5 EL Butter
	\item 1 EL Fischsoße
	\item 1 TL MSG
	\item Pfeffer
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Pilze kleinschneiden
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Große: dritteln
		\item $\rightarrow$ Kleine: halbieren
	\end{description}
	\item Zwiebel kleinschneiden
	\item Halme der Frühlingszwiebeln in große Röhrchen schneiden
	\item Kopf der Frühlingszwiebel seitlich schneiden
	\item Butter in einer heißen Pfanne erhitzen
	\item Sobald es gut riecht: Zwiebel rein
	\item Pilze rein
	\item Nach kurzer Zeit: Fischsoße rein
	\item Gegen Ende: MSG und Frühlingszwiebeln rein
	\item Pilze mit Pfeffer bestreuen
\end{enumerate}
