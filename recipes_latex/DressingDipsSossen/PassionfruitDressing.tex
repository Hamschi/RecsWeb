\subsection{Passionfruit Dressing}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 3 Passionsfruits
	\item 2 TL Aprikosengelee
	\item 1 TL Senf
	\item 3 EL Balsamico Essig (weiß)
	\item 50 ml Olivenöl
	\item 1 Prise Salz
	\item 1 Prise Cayennepfeffer
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Fruchtfleisch mit Hilfe eines Siebes in einen Mixer streichen
	\item Gelee, Senf und Essig zum Fruchtfleisch hinzufügen, mixen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Währenddessen das Öl in einem dünnen Strahl einlaufen lassen
	\end{description}
	\item Dressing mit Salz und Cayenne abschmecken
\end{enumerate}
