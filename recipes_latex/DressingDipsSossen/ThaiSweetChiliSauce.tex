\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item Chili (nach Schärfetoleranz)
	\item Definitiv auch große Chilis für Farbe (sind halt nicht scharf)
	\item 3-6 Zehen Knoblauch
	\item 150ml Wasser
	\item 100g Zucker
	\item 15ml Weißweinessig
	\item 15ml Fischsoße
	\item 1 Prise Salz
	\item 1 TL Kartoffelstärke (gehäuft)
	\item 5ml Wasser
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Chili grob schneiden 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Kerne entfernen, wenn man es weniger scharf möchte
	\end{description}
	\item Knoblauch und Chili pürieren
	\item Zucker, Wasser, Weißweinessig, Salz, Fischsoße, Chilimix in einem Topf sammeln und erhitzen
	\item Zum kochen bringen 
	\item Zum Köcheln bringe bis die Chili und der Knoblauch weich werden und der Zucker sich aufgelöst hat 
	\item Kartoffelstärke und Wasser vermengen
	\item Kartoffelstärkewasser hinzugeben und unterrühren
\end{enumerate}
