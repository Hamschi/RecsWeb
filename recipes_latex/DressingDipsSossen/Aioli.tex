\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 3-5 Knoblauchzehen
	\item 70ml Milch (3.5\% Fett)
	\item Salz
	\item 100ml Sonnenblumenöl
	\item 50ml Olivenöl (alternativ: Sonnenblumenöl)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Knoblauch schälen und grob vorschneiden
	\item Knoblauch, Milch und Salz in ein Mixgefäß geben
	\item Knoblauch mit der Milch aufpürieren
	\item Nach und nach unter ständigem Laufen des Mixers das Öl dazugeben
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Immer mit dem Pürierstab hoch und runter gehen
		\item \rarrow Erst ab einer bestimmten Menge Öl erhält die Aioli ihre Konsistenz
		\item \rarrow Wenn die Aioli zu dickflüssig ist: Milch hinzugeben
	\end{description}
	\item 
\end{enumerate}
