\selectlanguage{ngerman}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 3 EL Wasser
	\item 80g Zucker
	\item 3 EL Wasser
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Zucker gefolgt von 3 EL Wasser in einen kleinen Topf geben
	\item Das Zucker-Wasser Gemisch bei mittlerer Hitze zum Kochen bringen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Wenn der Zucker nicht gleichmäßig gold wird, reicht es, den Topf zu rotieren, Rühren ist nicht nötig
	\end{description}
	\item Sobald der Zucker gold-rot wird: die restlichen 3 EL Wasser dazugeben
	\item Solange kochen, bis kein Zucker mehr am Boden klebt und es zu einer dickflüssigen Masse wird
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ man kann einen Tropfen der Flüssigkeit in ein Glas geben. Wenn sich dieser Tropfen nicht auflöst und sich zu einer Kugel formt, ist alles richtig
		\item $\rightarrow$ Flüssigkeit sollte weder hart noch klebrig sein
	\end{description}
	\item Herd nun ausmachen und warten, bis die Flüssigkeit nicht mehr kocht
\end{enumerate}
