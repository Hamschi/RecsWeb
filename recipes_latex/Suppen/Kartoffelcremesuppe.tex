\subsection{Kartoffelcremesuppe (need testing)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 400g Kartoffeln
	\item 700ml Wasser
	\item 1 Stck. Suppenwürze (nach Grundrezept)
	\item 1 Becher Schlagsahne
	\item 1 Prise Salz
	\item 1 Prise Pfeffer
	\item 1 Prise Kümmel
	\item 1 Bund Petersilie
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Kartoffel schälen und in grobe Würfel schneiden
	\item Kartoffeln in einem großen Topf mit dem Wasser 15 Minuten bei mittlerer Hitze köcheln lassen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ mit Suppenwürze würzen
	\end{description}
	\item Kartoffeln mit dem Stabmixer pürieren und mit der Schlagsahne aufgießen
	\item dann mit Salz, Pfeffer und Kümmel abschmecken
	\item die Petersilie, waschen, trockenschütteln, feinhacken und in die Suppe rühren
\end{enumerate}
