\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500ml Wasser
	\item 1 TL Salz
	\item 400g Kartoffeln
	\item 300g Karotten
	\item 2 Stangen Sellerie
	\item 10-15 Schalotten
	\item 40g Knoblauch
	\item 1 Zwiebel 
	\item 1 Handvoll Petersilie
	\item 1kg Rindfleisch
	\item 100g Shimiji-Pilze (andere Pilze gehen auch)
	\item Butter
	\item 2 EL Chili Soße
	\item 200g Tomaten (passiert)
	\item 3 EL Austernsoße
	\item 120ml Rotwein
	\item 1.5 EL Hähnchenbrühepulver
	\item 1 TL Knoblauchpulver
	\item 4 EL Zucker
	\item 1 TL Pfeffer 
	\item 1/2 TL Salz
	\item 700ml Wasser
	\item 1 EL Chilisoße
	\item 10g Butter
	\item 40ml Wein 
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Salz in das Wasser geben
	\item Karotten und Sellerie in 5cm lange Stangen schneiden (ca. 1.5cm breit)
	\item Kartoffeln schälen und in mundgerechte Stücke schneiden
	\item Kartoffeln und Karotten in das Wasser geben und darin waschen
	\item Zwiebel in Wedges schneiden
	\item Petersilie klein hacken, Pilze dritteln
	\item Knoblauch und Schalotten klein hacken und pürieren
	\item Fleisch etwas größter als mundgerecht schneiden
	\item In einer Pfanne Öl und 2 EL Butter erhitzen 
	\item Fleisch reingeben und mit Knoblauchpulver, Salz und Pfeffer bestreuen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ ermöglicht womöglich mehrere Durchgänge, um alles zu braten
	\end{description}
	\item Salz, Hühnerbrühepulver, Zucker, Austernsoße, Pfeffer, Tomaten (120g, passiert), Chilisoße zu einer Soße vermengen
	\item In einem Topf 40g Butter erhitzen und darin das Pürierte braun braten
	\item Danach: Die Soße dazugeben und umrühren
	\item 80ml Rotwein dazugeben und umrühren
	\item Fleisch dazugeben und umrühren
	\item Nach ca. 5 Minuten: 80g Tomaten (passiert) in eine Schüssel tun
	\item Wasser dazugeben und alles umrühren 	
	\item Tomatenwasser in den Topf geben und zum köcheln bringen bis das Fleisch weich wird (ca. 40 Minuten)
	\item Kartoffeln und Karotten ca. 3 Minuten anbraten und zur Seite legen
	\item Zwiebel und Sellerie nun in der gleichen Pfanne anbraten
	\item Nach ca. 2 Minuten: Pilze mitbraten
	\item Sobald das Fleisch weich ist: Kartoffeln und Karotten hinzugeben
	\item Chilisoße hinzugeben und weitere 5-7 Minuten köcheln lassen
	\item Danach: Pilze, Zwiebeln und Sellerie hinzugeben und den Herd ausschalten
	\item Mit Petersilie garnieren
\end{enumerate}
