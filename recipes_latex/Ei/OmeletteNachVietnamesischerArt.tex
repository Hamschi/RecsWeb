\subsection{Omelette nach vietnamesischer Art}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 3 Eier
	\item 3 Halme Frühlingszwiebel
	\item 1 EL Fischsoße
	\item MSG
	\item Pfeffer
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Eier schlagen
	\item Frühlingszwiebel kleinschneiden
	\item Gewürze zum Ei geben und verrühren
	\item Frühlingszwiebel zum Ei geben und verrühren
	\item Herdstufe 9
	\item Ei auf der Pfanne braten bis das Ei nicht mehr flüssig ist
	\item Ei wenden
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Wenn das Wenden zu schwer sein sollte, dann kann man das Ei halbieren/ vierteln
	\end{description}
	\item sobald beide Seiten nicht mehr flüssig sind: fertig
\end{enumerate}
