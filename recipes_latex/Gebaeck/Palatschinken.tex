\subsection{Palatschinken (5 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 250g Mehl
	\item 2 Eier
	\item 1/2l Milch
	\item 1 Schuss Öl (alternativ Butter für die Pfanne)
	\item 1 Prise Salz
	\item Zum Bestreuen: Marmelade, Puderzucker, Nutella, Käse und Schinken...
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Mehl, Milch, Eier und das Salz mit einem Schlagbesen in einer Schüssel glattrühren
	\item 10 Minuten ziehen lassen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ sollte der Teig zu dick sein: mit Mineralwasser verdünnen
		\item $\rightarrow$ Unclear: \url{https://www.gutekueche.at/einfaches-palatschinkenrezept-rezept-4278}
	\end{description}
	\item in einer beschichteten Pfanne Öl ganz heiß erhitzen
	\item gleichmäßig eine Suppenkelle Teig in der Pfanne verteilen
	\item den Teig solange braten, bis dieser auf beiden Seiten goldbraun ist
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ den Palatschinken dabei mehrfach wenden
	\end{description}
	\item die fertigen Palatschinken im Ofen warmhalten (ca. 60\textdegree C)
	\item die Palatschinken mit beliebigen Zutaten bestreichen/ belegen und zusammenrollen
\end{enumerate}
