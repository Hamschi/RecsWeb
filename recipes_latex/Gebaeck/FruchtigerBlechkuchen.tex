\subsection{Fruchtiger Blechkuchen (need testing)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 250g Butter
	\item 175g Zucker
	\item 5 Eier
	\item 250g Mehl
	\item 1 TL Backpulver
	\item 1 Pck. Vanillezucker
	\item 1l Orangensaft
	\item 3 Pck. Vanillepuddingpulver
	\item 9 EL Zucker
	\item 1 Pck. Butterkekse
	\item etwas Orangensaft (zum Tränken d. Kekse)
	\item 2 Becher Schlagsahne
	\item bisschen Schokolade (zum Dekorieren)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Teig herstellen: Butter, Zucker, Ei, Mehl, Backpulver, Vanillezucker
	\item Teig gleichmäßig auf ein gebuttertes Blech verteilen
	\item 200\textdegree Unter-/Oberhitze (15 Minuten)
	\item Vanillepudding nach Anweisung zubereiten
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ statt Wasser/Milch Orangensaft benutzen
	\end{description}
	\item Blech mit kaltem Wasser füllen und Topf daraufstellen
	\item Pudding kontinuierlich rühren und regelmäßig das Wasser wechseln
	\item Pudding gleichmäßig auf dem gekühlten Kuchenboden verteilen
	\item Butterkekse in Orangensaft tauchen und gleichmäßig auf den Kuchen verteilen
	\item Sahnesteif zur Schlagsahne hinzufügen
	\item Schlagsahne steifschlagen
	\item Sahne gleichmäßig auf die Kekse verteilen
	\item Schokolade m.H. eines Messer in kleine Streusel schneiden
	\item Streusel auf den Kuchen verteilen
\end{enumerate}
