\subsection{Erdnussbutter-Kekse (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 70g Erdnussbutter
	\item 70g Zucker
	\item 1 Ei
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Ei und Zucker verrühren, bis sich der Zucker aufgelöst hat
	\item die Erdnussbutter dazurühren, bis ein glatter Teig entstanden ist
	\item aus der Masse kleine Kugeln formen
	\item die Kugeln auf ein mit Backpapier ausgelegtes Ofenblech legen und mit einer Gabel flachdrücken
	\item die Kekse bei 175 \textdegree C für 6-8 Minuten backen (Umluft)
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ fertig, sobald die Ränder leicht bräunen
	\end{description}
\end{enumerate}
