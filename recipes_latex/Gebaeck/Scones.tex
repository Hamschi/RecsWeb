\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 450g Mehl
	\item 113g Butter
	\item 115g Zucker
	\item 4 TL Backpulver
	\item 1 Ei
	\item 170-285ml Milch
	\item 1 Eigelb (als Glasur)
	\item Füllung: Butter, Marmelade, Clotted Cream,...
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Ofen auf 180\textdegree C einstellen
	\item Butter in kleine Würfel schneiden 
	\item Mehl, Backpulver und Zucker in einer großen Schüssel verrühren
	\item Butter einmassieren bis eine Textur entsteht, die Brotkrümmel ähnelt
	\item Füge das Ei und die Milch hinzu und verrühre diese Zutaten leicht mit der Hand 
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Schütte nicht sofort die ganze Milch rein, gehe es etwas langsamer an
		\item \rarrow Knete nicht diesen Teig!
	\end{description}
	\item Sobald ein Teig entsteht: Bestreue die Arbeitsoberfläche mit Mehl und knete den Teig leicht, sodass eine glatte Oberfläche entsteht
	\item Rolle den Teig aus (Dicke ca. 2.5cm)
	\item Schneide runde Formen aus diesem Teig 
	\item Streiche Eigelb über die Scones
	\item Backe die Scones für 15 Minuten bis sie goldbraun sind
	\item Schneide die Scones auf und tue Füllung rein
\end{enumerate}
