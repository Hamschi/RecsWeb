\subsection{Muffins mit Schokostückchen (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 250g Butter (weich)
	\item 200g Zucker (noch nicht reduziert)
	\item 1 Prise Salz
	\item 1 Pck. Vanillezucker
	\item 4 Eier
	\item 300g Mehl
	\item 1 Pck. Backpulver
	\item 220g Schokotropfen
	\item 125ml Milch
	\item Puderzucker zum Dekorieren (optional)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Backofen auf 200 \textdegree C Ober-/Unterhitze vorheizen
	\item in einer Schüssel die Eier schaumig rühren
	\item Butter, Zucker, Salz und Vanillezucker gut mit dem Mixer darin verrühren bis eine cremige Masse entsteht (bin mir hier nicht sicher, ob getrennt oder mit Ei)
	\item Milch zufügen und weiter mixen
	\item das Mehl und das Backpulver vermischen und in den Teig unterrühren
	\item die Schokotropfen hinzugeben und mit dem Teig gut vermischen
	\item den Teig gleichmäßig in die Muffinformen füllen 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Füllhöhe der Muffinform: ca. 2/3 der Vertiefung
	\end{description}
	\item im vorgeheizten Ofen 15 Minuten backen
	\begin{description}[noitemsep]
		\item $\rightarrow$ wenn kein feuchter Teig mehr an dem Stäbchen klebt, dann sind die fertig
	\end{description}
\end{enumerate}
