\subsection{Cannelé (12 Stück)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500ml Milch
	\item 180g Zucker
	\item 2 Eier
	\item 2 Eigelb
	\item 1/2 Flasche Rumaroma
	\item 1 Pck. Vanillezucker
	\item 100g Mehl
	\item 40g Butter
	\item 1 Prise Salz
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Zucker, Mehl und Salz miteinander vermengen
	\item Eier und Eigelb mit einer Gabel leicht verquirlen und dazugeben
	\item Milch mit dem Vanillezucker und der Butter aufkochen lassen 
	\item Danach: Milch langsam zur Zucker-Mehl-Mischung dazugeben
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ dabei ständig rühren
	\end{description}
	\item Sobald Teig lauwarm: Rumaroma dazugeben
	\item Teig für 2-3 Tage im Kühlschrank aufbewahren
	\item Danach: Teig rühren/schlagen bis sich die Butter wieder aufgelöst 
	\item Backformen mit Butter einfetten und eine Prise Zucker an den Rand streuen
	\item Formen bis zu 2/3 mit dem Teig füllen und auf ein Gitter platzieren
	\item Im auf 240\textdegree C vorgeheizten Ofen 10 Minuten backen
	\item Hitze senken auf 190\textdegree C und weitere 30-35 Minuten backen
	\item kühlen lassen und aus der Backform holen	
\end{enumerate}
