\subsection{Erdbeerkuchen nach Mamas Art (need pic)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 100g Butter
	\item 100g Zucker
	\item 2 Eier
	\item 100g Mehl
	\item 1 Pck. Vanillezucker
	\item 1 TL Backpulver
	\item 1 Prise Salz
	\item 2 EL Milch 
	\vspace{0.5cm}
	\item 1 Becher Schlagsahne
	\item 1 EL Zucker
	\item 500g Erdbeeren
	\item 1 Pck. Sahnesteif
	\item 1 Pck. Tortenguss
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Teig herstellen: Butter, Zucker, Ei 
	\item Mehl, Backpulver, Vanillezucker, Salz
	\item Milch
	\item Form für den Kuchenboden einfetten
	\item Teig in die Form geben
	\item 180\textdegree Umluft für 15-20 Minuten
	\item Danach: Teig abkühlen lassen
	\item Währenddessen: Erdbeeren schneiden (halbieren)
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ ggf. mit einem EL Zucker versüßen
	\end{description}
	\item Zucker in die die Schlagsahne geben 
	\item Schlagsahne zubereiten
	\item Sahne auf der fertigen Boden verteilen
	\item Erdbeeren ordentlich auf den Kuchen platzieren
	\item Glasur nach Anweisungen der Verpackung zubereiten
	\item Glasur über den Kuchen schütten
	\item Kuchen kühlen lassen
\end{enumerate}
