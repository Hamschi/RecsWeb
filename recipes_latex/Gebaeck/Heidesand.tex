\subsection{Heidesand (need testing)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 250g Butter
	\item 250g Zucker
	\item 1 Pck. Vanillezucker
	\item 1 Prise Salz
	\item 375g Mehl
	\item Zucker zum Wälzen (war nicht in Babsis Rezept)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Butter zerlassen, bräunen und abkühlen lassen
	\item Danach: die Butter mit den anderen Zutaten zu Knetteig verarbeiten
	\item Rollen mit etwa 4cm Durchmesser formen
	\begin{description}[noitemsep,nolistsep]
		\item $\rightarrow$ in vielen Rezepten werden diese Rollen in Frischhaltefolie gelegt
	\end{description}
	\item Teig mehrere Stunden im Kühlschrank ruhen lassen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ am besten über Nacht
	\end{description}
	\item Danach (nicht in Babsis Rezept enthalten): die Rollen in Zucker wälzen 
	\item 1/2 cm dicke Scheiben abschneiden
	\item die Scheiben auf dem ungefetteten Blech 25-30 Minuten bei 170-180\textdegree C backen
\end{enumerate}
