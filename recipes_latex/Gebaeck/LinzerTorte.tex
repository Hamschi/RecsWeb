\subsection{Linzer Torte}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 150g Butter
	\item 150g Staubzucker
	\item 250g Mehl(700)
	\item 100g Haselnüsse 
	\item 1 Ei
	\item 10g Backpulver
	\item 1 Prise Vanilleschote (Mark ausgekratzt)
	\item 1 Msp. Zitronenschale (abgerieben)
	\item 1/2 TL Zimt (fein gemahlen)
	\item 1 Msp. Gewürznelkenpulver
	\item 120g Mandeln (gehobelt)
	\item 300g Johannisbeermarmelade 
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Butter und Zucker verkneten
	\item Mehl sieben und mit Backpulver mischen
	\item Eier, Mehl, Nüss, Zimt, Gewürznelkenpulver, Zitronenabrieb und das Mark der Vanilleschote dazugeben
	\item alles rasch und kräftig verkneten
	\item den fertigen Teig zu einer Kugel formen, in Frischhaltefolie wickeln und in den Kühlschrank stellen
	\item nach einiger Zeit aus dem Kühlschrank nehmen und den Teig in vier gleichgroße Teile unterteilen
	\item 3/4 des Teiges auf ca. 1,5cm ausrollen
	\begin{description}[noitemsep,nolistsep]
		\item $\rightarrow$ Tortendurchmesser ca. 24cm
	\end{description}
	\item die Marmelade aufstreichen
	\item den restlichen Teig zu Rollen formen und dasa Gitter und Rand auf die Marmelade auflegen
	\item das Gitter mit Ei bestreichen und am Rand mit gehobelten mandeln bestreuen
	\item Danach: die Torte für 40-45 Minuten bei ca. 180\textdegree C Ober-/Unterhitze in den vorgeheizten Backofen geben 
\end{enumerate}
