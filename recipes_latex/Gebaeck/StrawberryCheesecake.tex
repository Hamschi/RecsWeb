\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 200g Digestive Kekse
	\item 150g Butter
	\item 500g Philadelphia-Frischkäse
	\item 300g Sahne
	\item 100g Joghurt (3.5\% Fett)
	\item 70g Zucker
	\item 3 Blatt Gelatine 
	\item Zitronenabrieb
	\item 1 TL Vanillepaste (alternativ: Tropfen Vanille-Extrakt oder Schote)
	\item 350g Erdbeeren
	\item 20g Zucker
	\item 5 Blatt Gelatine
\end{itemize}


\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Kekse in einer Tüte zerkleinern
	\item Butter schmelzen und mit den Keksen mischen
	\item Eine 28cm-Kuchenform mit Backpapier auslegen und die Keksmasse gleichmäßig darauf verteilen und für 15 Minuten in den Kühlschrank stellen
	\item 3 Gelatine in Wasser einweichen
	\item Sahne mit dem Zucker steif schlagen und kalt stellen
	\item Philadelphia mit dem Vanillemark, dem Zitronenabrieb und Joghurt kurz glatt rühren
	\item Einen kleinen Teil der Sahne aufheben, um ihn mit der Gelatine zu vermischen
	\item Rest der Sahne nach und nach in den Philadelphia unterheben
	\item Die Gelatine in einem Topf erhitzen und flüssig werden lassen
	\item Die zurückbehaltene Sahne mit der Gelatine vermischen
	\item Diese Mischung in die Sahne-Philadelphia-Mischung geben und unterheben + unterrühren
	\item Masse gleichmäßig auf den Kuchenboden verteilen und 3-4h kalt stellen
	\item Restliche Gelatine einweichen
	\item Erdbeeren waschen und mit dem Zucker auf dem Herd zum Kochen bringen, 2 Minuten köcheln lassen und von der Hitze nehmen
	\item Eingeweichte Gelatine zu den Erdbeeren geben und mit einem Mixstab alles kurz pürieren
	\item Masse durch einen Sieb geben und auf einem Wasserbad abkühlen lassen
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Nicht zu kalt, sonst wird es fest
	\end{description}
	\item Erdbeermasse auf dem Kuchen verteilen und alles für 3-4h kalt stellen
\end{enumerate}
