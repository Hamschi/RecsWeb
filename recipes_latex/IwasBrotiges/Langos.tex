\paragraph{Zutaten}
\begin{itemize}[noitemsep]	
	\item 2-3 Zehen Knoblauch
	\item 50ml  Öl
	\item 50ml Wasser
	\item 1 TL Salz
	\vspace{0.5cm}
	\item 500g Mehl
	\item 350ml Wasser (lauwarm)
	\item 2 TL Zucker
	\item 20g Hefe (frisch)
	\item 1 EL Salz
	\item 100ml Öl
	\item Öl zum Frittieren
	\item Topping: Sour Cream, Käse (Cheddar, Gouda,...)
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Knoblauch in ein Glas zerdrücken
	\item Wasser,  Öl und Salz dazugeben und gut durchmischen
	\item Glas in den Kühlschrank stellen	
	\item lauwarmes Wasser in eine Schüssel schütten
	\item Zucker darin auflösen lassen
	\item Hefe darin nun auflösen lassen
	\item Mehl in eine große Schüssel sieben
	\item Mehl mit dem Salz vermengen und anschließend das Hefewasser hinzugeben
	\item Alles zu einem Teig für ca. 10 Minuten verkneten bis ein glatter, elastischer Teig entsteht und dieser Teig nicht mehr an der Schüssel klebt
	\item Den Teig abdecken und für 20 Minuten sich ausweiten lassen
	\item Öl um den Teig gießen
	\item Den Teig ziehen, sodass das Öl unter den Teig gelingt
	\item Den Teig wieder für 5-8 Minuten abgedeckt lassen
	\item Eine Handvoll Teig nehmen und in eine runde Form formen (sollte für ca. 6 Stücke ausreichen)
	\item Die Teige für weitere 10 Minuten ruhen lassen
	\item Öl sehr heiß erhitzen
	\item Einen Teigball nehmen und mit der Hand ausweiten und zu einer einer flachen, runden Form formen
	\begin{description}[noitemsep, nolistsep]
		\item \rarrow Außen sollte der Teig dicker sein als Innen
	\end{description}
	\item Lege den Teig vorsichtig ins Öl
	\item Sobald der Teig goldbraun ist: wenden und für ca. 1 Minute frittieren
	\item Diese Schritte mit den restlichen Teigen wiederholen
	\item Langos mit dem Knoblauchöl bestreichen und 2-3 EL Schmand darauf verteilen und mit Käse bestreuen
\end{enumerate}
