\subsection{Naan (8 Stück)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 300g Weizenmehl (Typ 405)
	\item 3 EL Joghurt
	\item 1 EL Öl
	\item 2 EL Wasser (warm)
	\item 1 TL Zucker
	\item 1 TL Salz
	\item 1/2 TL Hefe
	\item 1/4 TL Backpulver
	\item 130ml Milch (nach Bedarf)
	\item etwas Butter
\end{itemize}

\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Zucker und Hefe im Wasser auflösen lassen
	\item Mehl, Salz, Backpulver und Öl in einer Schüssel vermengen
	\item Hefemischung und Joghurt in die Mehlmischung geben
	\item Nach und nach Milch dazugeben und kneten
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Von der Konsistenz her darf der Teig weich sein
	\end{description}
	\item Teig zudecken und 1 Stunde gehen lassen
	\item Teig wieder kneten und ebenfalls in 10 Teile teilen
	\item Eine Teigkugel nehmen (bedecke die restlichen Kugeln mit einem Tuch) und ausrollen
	\item Pfanne richtig heiß werden lassen und Naan hinein legen
	\item ein Ofengitter vorbereiten und auf einer Herdplatte warm werden lassen
	\item Sobald das Naan einseitig fertig ist: die andere Seite auf dem Gitter fertig backen (in der Pfanne geht aber auch)
	\item Naan am Ende mit Butter bestreichen
\end{enumerate}
