\subsection{Pizza (2-3 Pizzen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500g Mehl Type 550 (Strong Flour)
	\item 1 TL Salz
	\item 1 Pck. trockene Hefe (7g)
	\item 325ml Wasser (warm)
	\item Brotkrümmel  
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Hefe im Wasser auflösen lassen 
	\item das Salz ins Mehl mischen und anschließend die Hefeflüssigkeit mit dem Mehl vermengen und zu kneten
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ ggf. benötigt man mit Mehl bestaubte Hände
		\item $\rightarrow$ den Teig an einem Ende mit der Hand halten und mit der anderen Hand das andere Ende greifen und von einem wegziehen
		\item und dann wieder zu sich ziehen 
	\end{description}
	\item großzügig Mehl auf die Bearbeitungsoberfläche geben und den Teig darin wälzen lassen
	\item Teig mit einem Messer in vier gleichgroße Stücke zerteilen und diese abrunden 
	\item Backblech mit Brotkrümmel bestreuen und die Kugeln mit großem Abstand zueinander auf das Backblech legen 
	\item diese mit Mehl bestreuen und den Teig mindestens 2h ziehen lassen
	\item eine Kugel auf die mit Mehl bestreute Arbeitsoberfläche legen, mit der Hand flachdrücken, drehen und nochmal flachdrücken 
	\item Teig flachdingsen und dann Tomatensauce darauf verteilen, Olivenöl darübernieseln lassen
	\item Beläge etc. drauflegen und am Ende mit einer Prise Brotkrümmel bestreuen 
	\item für 220 \textdegree C 7 Minuten lang backen lassen (Ober- Unterhitze)
\end{enumerate}
