\subsection{Khachapuri (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 250g Brotmehl (strong white bread flour)
	\item 160ml 23\textdegree heißes Wasser (alternativ: Milch)
	\item 1 TL Salz
	\item 1 TL Zucker
	\item 3g Trockenhefe (alternativ: 9g frische Hefe)
	\item 10g Olivenöl (alternativ: Butter)
	\item 100g Feta (zerbröselt)
	\item 100g Mozzarella (gerieben)
	\item 100g Cheddar (gerieben)
	\item 1 Ei
	\item 1 EL Mehl
	\item 2 Eier (das meiste Eiweiß entfernen)
	\item Butter
	\item Petersilie (gehackt) (optional)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item In einer Schüssel Wasser, Hefe, Zucker vermischen und die Trockenzutaten auflösen lassen
	\item Mehl und Salz mit der Flüssigkeit vermengen 
	\item Solange kneten, bis kein trockenes Mehl mehr zu sehen ist 
	\item Sobald Teig glattgeknetet ist: zu einer Kugel formen
	\item Teig 45 Minuten ziehen lassen 
	\item Danach: Teig entnehmen, die glatte Seite auf den Boden legen und falten
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ dabei im Kreis die Seiten des Teiges nehmen und in die Mitte falten
		\item $\rightarrow$ solange, bis man wieder am Anfang angelangt ist
	\end{description}
	\item Teig dann wieder \glqq fest rollen\grqq
	\item Teig mit der glatten Seite nach oben wieder in die Schüssel tun und weitere 45 Minuten ziehen lassen
	\item Danach: Teig halbieren und beide Hälften falten und dann zu einer Kugel formen
	\item Teigkugeln in einem mit Mehl bestaubten, kleinen Backblech 30 Minuten ziehen lassen
	\item Währenddessen: Ofen vorheizen auf 230\textdegree C
	\item die Käsesorten mit dem Ei und dem Mehl mischen bis sie eine gleichmäßige Masse bilden
	\item Arbeitsfläche mit Mehl bestauben und die eine Hälfte Ovalig-flach ausrollen (ca. 0.5cm dick)
	\item Danach: an den länglichen Seiten 1cm vom Rand nehmen und über den Teig überlappen/ falten (insg. 3x auf jeder Seite)
	\item die Seiten zusammentwisten und somit verschließen
	\item das Brot-Bot auf ein Backpapier legen und die Hälfte der Füllung gleichmäßig darin verteilen
	\item Brot auf eine Backofen feste Pfanne legen und 13 Minuten im Ofen lassen
	\item mit einer Gabel oder einem Löffel ein kleines Loch rechts in die Füllung reinbohren und das Ei dort reinlegen
	\item 2 Minuten im Ofen backen
	\item danach die Pfanne aus dem Ofen entnehmen und den Rand mit geschmolzener Butter bestreichen
	\item ein kleine Flocke Butter auf die linkere Seite der Füllung legen
	\item ggf. das Khachapuri mit Petersilie bestreuen
\end{enumerate}
