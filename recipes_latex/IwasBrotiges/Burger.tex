\subsection{Burger (2 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 2 Buns
	\item 2 Patties
	\item Salz
	\item Pfeffer
	\item Öl 
	\item 2 Scheiben Cheddar
	\item 4 Scheiben Tomaten
	\item 1 Zwiebel (klein, weiß)
	\item 1 Gewürzgurke
	\item 4 2 EL Ketchup
	\item 2 EL Butter
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item die Patties auf beiden Seiten kräftig mit Salz und Pfeffer würzen 
	\item Patties auf beiden Seiten mit Öl \glqq{}bepinselns\grqq{}
	\item Patties in die heiße Pfanne legen und für 2-3 Minuten braten
	\item Währenddessen: Zwiebel in Zwiebelringe, Tomate und Gewürzgurke in Scheiben schneiden 
	\item Nach den 2-3 Minuten: Fleisch auf der anderen Seite anbraten 
	\item 1 EL Butter zur Pfanne hinzugeben und darin die Zwiebelringe goldbraun rösten
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Zwiebeln hin und wieder drehen und auch mit Salz würzen
	\end{description}
	\item gegen Ende Gewürzgurke dazugeben 
	\item Nach den 2 Minuten: Käse auf das Patty gefolgt von den Zwiebeln
	\item Sobald die Gurken \glqq{}latschig\grqq{} sind: auf den Patty legen
	\item Patty inkl. Belag aus der Pfanne nehmen
	\item 1 EL Butter in die Pfanne geben und darin die Bun-Hälften anrösten 
	\item Mit dem Burger zum Schluss die Pfanne auswischen 
	\item Die Bun-Innenseiten mit Ketchup beschmieren, zwei Tomatenscheiben drauf und dann den Rest
\end{enumerate}
