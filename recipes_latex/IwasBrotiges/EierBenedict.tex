\subsection{Eier Benedict}
\begin{itemize}[noitemsep]
	\item 500ml Wasser
	\item 3 EL Essig
	\item Speck
	\item 1 Toastbrötchen
	\item 4 EL Sauce Hollandaise
	\item 2 Eier
	\item Schnittlauch (zum Garnieren)
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Topf mit Wasser und Essig zum Kochen bringen
	\item Währenddessen: Speck in einer Pfanne anbraten, Toastbrot toasten und wenn möglich Petersilie kleinschneiden
	\item ein Ei in eine Tasse aufschlagen und vorsichtig ins ins Essigwasser gleiten lassen
	\item Mit Hilfe eines Löffels das Eiklar vorsichtig um den Eidotter ummanteln
	\item  sobald das Wasser an der Wasseroberfläche schwimmt, mit einer Schöpfkelle aus dem Wasser heben 
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Vorgang 3-5 mit dem anderen Ei wiederholen	
	\end{description}
	\item Aufbau: 1. Brötchen, 2. Speck, 3. Ei, 4. Sauce Hollandaise, 5. Petersilie
\end{enumerate}
