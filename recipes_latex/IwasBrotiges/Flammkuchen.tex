\subsection{Flammkuchen nach Elsässer Art (2 Stück)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 250g Mehl (Typ 550)
	\item 125ml Wasser
	\item 2 EL Olivenöl
	\item 5g Salz
	\vspace{0.5cm}
	\item 1 Becher Crème Fraîche (250g)
	\item 1 Zwiebel
	\item 200g Speck
	\item Frühlingszwiebeln (optional)
	\item Salz, Pfeffer
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Salz und Olivenöl im Wasser auflösen
	\item Mehl in die Flüssigkeit geben und verrühren
	\item Den Teig auf der Arbeitsfläche durchkneten bis ein schöner, geschmeidiger Teig entsteht (ca. 5 Minuten)
	\item Teig zu einer Kugel formen und in Frischhaltefolie abdecken
	\item Teig für 20-30 Minuten im Kühlschrank ruhen lassen
	\item Speck in feine Würfel schneiden
	\item Zwiebel in ganz dünne Scheiben schneiden
	\item Frühlingszwiebeln in ganz dünne Scheiben schneiden
	\item Crème Fraîche mit Salz und Pfeffer verrühren
	\item Sobald der Teig fertig geruht hat: Ofen auf 260\textdegree C vorheizen
	\item Teig in zwei Portionen unterteilen und jeweils ganz dünn ausrollen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Teig sollte nicht dicker als 1mm sein
	\end{description}
	\item Crème Fraîche, Zwiebeln und den Speck auf dem Teig verteilen
	\item Den Flammkuchen für 7 Minuten in den vorgeheizten Ofen
	\item Optional: Flammkuchen mit Frühlingszwiebeln garnieren
\end{enumerate}
