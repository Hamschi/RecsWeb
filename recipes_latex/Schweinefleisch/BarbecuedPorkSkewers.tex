\subsection{Barbecued Pork Skewer (8 Portionen)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep, nolistsep]
	\item 900g Schweineschulter
	\item 1/4 Gemüsezwiebel (sehr klein gehackt) 
	\item 1 Zehe Knoblauch (groß, gerieben)
	\item 1 EL Zucker (braun)
	\item 2 TL Salz
	\item 1 TL Pfeffer (schwarz)
	\item 1 TL Paprikapulver
	\item 1/2 TL Kreuzkümmel
	\item 1/4 TL Cayennepfeffer 
	\item Barbecue Soße
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Schulter längst halbieren und dann in möglichst dünne Scheiben schneiden
	\item lege das Fleisch in eine Schüssel und vermische es mit Händen mit Knoblauch, Zwiebeln, Zucker, Salz, Pfeffer, Paprikapulver, Kreuzkümmel, Cayennepfeffer 
	\item Fleisch mit Frischhaltefolie bedecken, sodass die Folie direkt das Fleisch überall abdeckt und für 4-12 Stunden in den Kühlschrank tun
	\item Fleisch durch Metallspieße spießen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ wenn ein Stück zu lang ist, dann muss es mehrfach auf den Spieß gespießt werden
		\item $\rightarrow$ möglichst eng und fest spießen
	\end{description}
	\item auf dem vorgeheizten Grill grillen und regelmäßig wenden, bis es von außen braun und von innen nicht mehr rosa ist
	\item nach ca. 30 Minuten das Fleisch mit Barbecue Soße bestreichen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ ca. weitere 10 Minuten
	\end{description}
\end{enumerate}
