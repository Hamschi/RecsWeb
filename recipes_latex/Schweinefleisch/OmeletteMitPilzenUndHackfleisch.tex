\subsection{Omelette mit Pilzen und Hackfleisch (not done yet)}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 3 Judasohren
	\item 1 Handvoll Glasnudeln
	\item 3 Eier
	\item 150g Hackfleisch
	\item 1/2 Zwiebel
	\item 1 TL Salz
	\item Pfeffer
	\item 1/2 TL MSG
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Pilze und Glasnudeln waschen und in heißes Wasser legen
	\item Zwiebel in kleine Würfel schneiden
	\item harten Kopf der Pilze abschneiden und in kleine Streifen schneiden
	\item Glasnudeln in kleine Nudeln schneiden
	\item Fleisch, Pilze, Nudeln, Eier und Gewürze in einer Schale mischen
	\item Pfanne mit Öl vorbereiten
	\item einen EL der Mischung nehmen und auf die Platte verteilen
	\item paar Minuten braten
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Vorgang wiederholen
	\end{description}
	\item Paar mal zwischendurch wenden
\end{enumerate}
