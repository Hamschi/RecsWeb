\subsection{Schweinerücken in Austernsoße (2 Portionen)}
\paragraph{Zubereitung}
\begin{itemize}[noitemsep]
	\item 500g Schweinerücken
	\item 3 EL Austernsoße
	\item 3 TL Fischsoße
	\item 1 TL Zucker (gehäuft)
	\item 1/2 TL Salz
	\item 1 TL Sojasoße (dunkel)
	\item 1/4 Zwiebel
	\item 1 bisschen Wasser
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Fett vom Fleisch trennen und entsorgen
	\item Fleisch in etwas größere Häppchen schneiden (ca. 1cm dick)
	\item alles in einem Topf geben und mit der Austersoße vermengen
	\item nun die Fischsoße dazugeben und wieder alles vermengen
	\item Zucker, Salz und die dunkle Sojasoße rein und wieder alles vermengen
	\item Fleisch mind. 1h ziehen lassen
	\item die Viertel Zwiebel in kleine Würfel schneiden 
	\item die ganze Zwiebel in lange Streifen schneiden
	\item die Zwiebelwürfel in einer Pfanne goldbraun anbraten
	\item Fleisch dazugeben und alles umrühren
	\item das Wasser zu dem Topf für die Marinade geben und das Wasser in die Pfanne schütten
	\item Nach 4 Minuten abschmecken und die Zwiebeln dazugeben und kurz braten
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ falls fad: mehr Austernsoße
	\end{description}
\end{enumerate}
