\subsection{Schweinespieße}
\paragraph{Zutaten}
\begin{itemize}[noitemsep]
	\item 500g Schweinebauch 
	\item 1/2 Zwiebel (lila)
	\item 1 Halm Zitronengras
	\item 2 EL Fischsoße
	\item 2 EL Austernsoße
	\item 1 EL Dunkle Sojasoße
	\item 1 EL Sesamöl
	\item 1/3 TL Salz
	\item 1/3 TL Zucker
	\item 1/3 TL MSG
	\item 1 Zwiebel (lila)
	\item 1 Zwiebel (gelb)
	\item Sesam zum Bestreuen
	\item 1 gehäufter TL Honig
\end{itemize}
\paragraph{Zubereitung}
\begin{enumerate}[noitemsep]
	\item Zwiebel (lila) und  Zitronengras kleinschneiden
	\item Zwiebel (lila) und Zitronengras zum Fleisch geben
	\item Gewürze zum Fleisch geben und kneten
	\item 30 Minuten ziehen lassen
	\item Zwiebel (lila) und Zwiebel (gelb) in flache Scheiben schneiden
	\item Abwechselnd Fleisch und Zwiebel mit einem Spieß durchstechen
	\begin{description}[noitemsep, nolistsep]
		\item $\rightarrow$ Reihenfolge. Zwiebel 1, Fleisch ohne Fett, Zwiebel 2, Fleisch mit Fett
	\end{description}
	\item Backofen: 230 \textdegree , Ober- und Unterhitze
	\item Währenddessen: Sesam anbraten und Honig mit Wasser verdünnen
	\item Sobald Spieß essbar aussieht (nicht schwarz): Spieße mit Honigswasser bestreichen und Sesam draufstreuen 
	\item Spieße wenden und Nummer 6 wiederholen
	\item weitere 2-3 Minuten drin lassen
\end{enumerate}
